130  {

134  Tilemap map = brushTarget.GetComponent<Tilemap>();
135  if (map == null)
136  return;
137 
138  int count = 0;
139  var listSize = position.size.x * position.size.y * position.size.z;
140  if (m_TileChangeDataList == null || m_TileChangeDataList.Capacity != listSize)
141  m_TileChangeDataList = new List<TileChangeData>(listSize);
142  m_TileChangeDataList.Clear();
143  foreach (Vector3Int location in position.allPositionsWithin)
144  {
145  Vector3Int local = location - position.min;

151  m_TileChangeDataList.Add(tcd);
152  count++;
153  }
154  // Duplicate empty slots in the list, as ExtractArrayFromListT returns full list
155  if (0 < count && count < listSize)
156  {
157  var tcd = m_TileChangeDataList[count - 1];
158  for (int i = count; i < listSize; ++i)
159  {
160  m_TileChangeDataList.Add(tcd);
161  }
162  }
163  var tileChangeData = NoAllocHelpers.ExtractArrayFromListT(m_TileChangeDataList);

231  public override void Rotate(RotationDirection direction, GridLayout.CellLayout layout)
232  {
233  switch (layout)
234  {
235  case GridLayout.CellLayout.Hexagon:
236  RotateHexagon(direction);
237  break;
238  case GridLayout.CellLayout.Isometric:
239  case GridLayout.CellLayout.IsometricZAsY:
240  case GridLayout.CellLayout.Rectangle:
241  {
242  Vector3Int oldSize = m_Size;
243  BrushCell[] oldCells = m_Cells.Clone() as BrushCell[];
244  size = new Vector3Int(oldSize.y, oldSize.x, oldSize.z);
245  BoundsInt oldBounds = new BoundsInt(Vector3Int.zero, oldSize);
246 
247  foreach (Vector3Int oldPos in oldBounds.allPositionsWithin)
248  {
249  int newX = direction == RotationDirection.Clockwise ? oldPos.y : oldSize.y - oldPos.y - 1;
250  int newY = direction == RotationDirection.Clockwise ? oldSize.x - oldPos.x - 1 : oldPos.x;
251  int toIndex = GetCellIndex(newX, newY, oldPos.z);
252  int fromIndex = GetCellIndex(oldPos.x, oldPos.y, oldPos.z, oldSize.x, oldSize.y, oldSize.z);
253  m_Cells[toIndex] = oldCells[fromIndex];
254  }
255 
256  int newPivotX = direction == RotationDirection.Clockwise ? pivot.y : oldSize.y - pivot.y - 1;
257  int newPivotY = direction == RotationDirection.Clockwise ? oldSize.x - pivot.x - 1 : pivot.x;
258  pivot = new Vector3Int(newPivotX, newPivotY, pivot.z);
259 
260  Matrix4x4 rotation = direction == RotationDirection.Clockwise ? s_Clockwise : s_CounterClockwise;
261  Matrix4x4 counterRotation = direction != RotationDirection.Clockwise ? s_Clockwise : s_CounterClockwise;
262  foreach (BrushCell cell in m_Cells)
263  {
264  Matrix4x4 oldMatrix = cell.matrix;
265  bool counter = (oldMatrix.lossyScale.x < 0) ^ (oldMatrix.lossyScale.y < 0);
266  cell.matrix = oldMatrix * (counter ? counterRotation : rotation);
267  }
268  }
269  break;
270  }
271  }

273  private static Vector3Int RotateHexagonPosition(RotationDirection direction, Vector3Int position)
274  {

292  private void RotateHexagon(RotationDirection direction)
293  {
294  BrushCell[] oldCells = m_Cells.Clone() as BrushCell[];
295  Vector3Int oldPivot = new Vector3Int(pivot.x, pivot.y, pivot.z);
296  Vector3Int oldSize = new Vector3Int(size.x, size.y, size.z);
297  Vector3Int minSize = Vector3Int.zero;
298  Vector3Int maxSize = Vector3Int.zero;
299  BoundsInt oldBounds = new BoundsInt(Vector3Int.zero, oldSize);
300  foreach (Vector3Int oldPos in oldBounds.allPositionsWithin)
301  {
302  if (oldCells[GetCellIndex(oldPos.x, oldPos.y, oldPos.z, oldSize.x, oldSize.y, oldSize.z)].tile == null)
303  continue;
304  var pos = RotateHexagonPosition(direction, oldPos - oldPivot);
305  minSize.x = Mathf.Min(minSize.x, pos.x);
306  minSize.y = Mathf.Min(minSize.y, pos.y);
307  maxSize.x = Mathf.Max(maxSize.x, pos.x);
308  maxSize.y = Mathf.Max(maxSize.y, pos.y);
309  }
310  Vector3Int newSize = new Vector3Int(1 + maxSize.x - minSize.x, 1 + maxSize.y - minSize.y, oldSize.z);
311  Vector3Int newPivot = new Vector3Int(-minSize.x, -minSize.y, oldPivot.z);
312  UpdateSizeAndPivot(newSize, new Vector3Int(newPivot.x, newPivot.y, newPivot.z));
313  foreach (Vector3Int oldPos in oldBounds.allPositionsWithin)
314  {
315  if (oldCells[GetCellIndex(oldPos.x, oldPos.y, oldPos.z, oldSize.x, oldSize.y, oldSize.z)].tile == null)
316  continue;
317  Vector3Int newPos = RotateHexagonPosition(direction, new Vector3Int(oldPos.x, oldPos.y, oldPos.z) - oldPivot) + newPivot;
318  m_Cells[GetCellIndex(newPos.x, newPos.y, newPos.z)] = oldCells[GetCellIndex(oldPos.x, oldPos.y, oldPos.z, oldSize.x, oldSize.y, oldSize.z)];
319  }
320  // Do not rotate hexagon cell matrix, as hexagon cells are not perfect hexagons
321  }

491  private static void FlipCells(ref BrushCell[] cells, Vector3 scale, bool skipRotation)
492  {
493  Matrix4x4 flip = Matrix4x4.TRS(Vector3.zero, Quaternion.identity, scale);
494  foreach (BrushCell cell in cells)
495  {
496  Matrix4x4 oldMatrix = cell.matrix;
497  if (skipRotation || Mathf.Approximately(oldMatrix.rotation.x + oldMatrix.rotation.y + oldMatrix.rotation.z + oldMatrix.rotation.w, 1.0f))
498  cell.matrix = oldMatrix * flip;
499  else
500  cell.matrix = oldMatrix * s_180Rotate * flip;
501  }
