105  // 1. Find top-most rectangle
106  // 2. Sweep it vertically to find out all rects from that "row"
107  // 3. goto 1.
108  // This will give us nicely sorted left->right top->down list of rectangles
109  // Works for most sprite sheets pretty nicely
110  private List<Rect> SortRects(List<Rect> rects)
111  {
112  List<Rect> result = new List<Rect>();
113 
114  while (rects.Count > 0)
115  {
116  // Because the slicing algorithm works from bottom-up, the topmost rect is the last one in the array
117  Rect r = rects[rects.Count - 1];
118  Rect sweepRect = new Rect(0, r.yMin, textureActualWidth, r.height);
119 
120  List<Rect> rowRects = RectSweep(rects, sweepRect);
121 
122  if (rowRects.Count > 0)
123  result.AddRange(rowRects);
124  else
125  {
126  // We didn't find any rects, just dump the remaining rects and continue
127  result.AddRange(rects);
128  break;
129  }
130  }
131  return result;
132  }

134  private List<Rect> RectSweep(List<Rect> rects, Rect sweepRect)
135  {
136  if (rects == null || rects.Count == 0)
137  return new List<Rect>();
138 
139  List<Rect> containedRects = new List<Rect>();
140 
141  foreach (Rect rect in rects)
142  {
143  if (rect.Overlaps(sweepRect))
144  containedRects.Add(rect);
145  }
146 
147  // Remove found rects from original list
148  foreach (Rect rect in containedRects)
149  rects.Remove(rect);
150 
151  // Sort found rects by x position
152  containedRects.Sort((a, b) => a.x.CompareTo(b.x));
153 
154  return containedRects;
155  }

157  private int AddSprite(Rect frame, int alignment, Vector2 pivot, AutoSlicingMethod slicingMethod, int originalCount, ref int nameIndex)
158  {
159  int outSprite = -1;
160  switch (slicingMethod)
161  {
162  case AutoSlicingMethod.DeleteAll:
163  {
164  while (outSprite == -1)
165  {
166  outSprite = AddSprite(frame, alignment, pivot, GenerateSpriteNameWithIndex(nameIndex++), Vector4.zero);
167  }
168  }
169  break;
170  case AutoSlicingMethod.Smart:

176  existingRect.rect = frame;
177  existingRect.alignment = (SpriteAlignment)alignment;
178  existingRect.pivot = pivot;
179  }
180  else
181  {
182  while (outSprite == -1)
183  {
184  outSprite = AddSprite(frame, alignment, pivot, GenerateSpriteNameWithIndex(nameIndex++), Vector4.zero);
185  }
186  }
187  }
188  break;
189  case AutoSlicingMethod.Safe:
190  {
191  outSprite = GetExistingOverlappingSprite(frame, originalCount);
192  while (outSprite == -1)
193  {
194  outSprite = AddSprite(frame, alignment, pivot, GenerateSpriteNameWithIndex(nameIndex++), Vector4.zero);
195  }
196  }
197  break;

202  private int GetExistingOverlappingSprite(Rect rect, int originalCount, bool bestFit = false)
203  {
204  var count = Math.Min(originalCount, m_RectsCache.spriteRects.Count);
205  int bestRect = -1;
206  float rectArea = rect.width * rect.height;
207  if (rectArea < kOverlapTolerance)
208  return bestRect;
209 
210  float bestRatio = float.MaxValue;
211  float bestArea = float.MaxValue;
212  for (int i = 0; i < count; i++)
213  {
214  Rect existingRect = m_RectsCache.spriteRects[i].rect;
215  if (existingRect.Overlaps(rect))
216  {
217  if (bestFit)
218  {
219  float dx = Math.Min(rect.xMax, existingRect.xMax) - Math.Max(rect.xMin, existingRect.xMin);
220  float dy = Math.Min(rect.yMax, existingRect.yMax) - Math.Max(rect.yMin, existingRect.yMin);
221  float overlapArea = dx * dy;
222  float overlapRatio = Math.Abs((overlapArea / rectArea) - 1.0f);
223  float existingArea = existingRect.width * existingRect.height;
224  if (overlapRatio < bestRatio || (overlapRatio < kOverlapTolerance && existingArea < bestArea))
225  {
226  bestRatio = overlapRatio;
227  if (overlapRatio < kOverlapTolerance)
228  bestArea = existingArea;
229  bestRect = i;
230  }
231  }
232  else
233  {
234  bestRect = i;
235  break;
236  }
237  }
238  }
239  return bestRect;
240  }

343  public IEnumerable<Rect> GetIsometricRects(Vector2 size, Vector2 offset, bool isAlternate, bool keepEmptyRects)
344  {
345  var textureToUse = GetTextureToSlice();
346  var gradient = (size.x / 2) / (size.y / 2);
347  bool isAlt = isAlternate;
348  float x = offset.x;
349  if (isAlt)
350  x += size.x / 2;
351  float y = textureToUse.height - offset.y;
352  while (y - size.y >= 0)
353  {
354  while (x + size.x <= textureToUse.width)
355  {
356  var rect = new Rect(x, y - size.y, size.x, size.y);
357  if (!keepEmptyRects)
358  {
359  int sx = (int)rect.x;
360  int sy = (int)rect.y;
361  int width = (int)size.x;
362  int odd = ((int)size.y) % 2;
363  int topY = ((int)size.y / 2) - 1;
364  int bottomY = topY + odd;
365  int totalPixels = 0;
366  int alphaPixels = 0;
367  {
368  for (int ry = 0; ry <= topY; ry++)
369  {
370  var pixelOffset = Mathf.CeilToInt(gradient * ry);
371  for (int rx = pixelOffset; rx < width - pixelOffset; ++rx)
372  {
373  if (PixelHasAlpha(sx + rx, sy + topY - ry, textureToUse))
374  alphaPixels++;
375  if (PixelHasAlpha(sx + rx, sy + bottomY + ry, textureToUse))
376  alphaPixels++;
377  totalPixels += 2;
378  }
379  }
380  }
381  if (odd > 0)
382  {
383  int ry = topY + 1;
384  for (int rx = 0; rx < size.x; ++rx)
385  {
386  if (PixelHasAlpha(sx + rx, sy + ry, textureToUse))
387  alphaPixels++;
388  totalPixels++;
389  }
390  }
391  if (totalPixels > 0 && ((float)alphaPixels) / totalPixels > 0.01f)
392  yield return rect;
393  }
394  else
395  yield return rect;
396  x += size.x;
397  }
398  isAlt = !isAlt;
399  x = offset.x;
400  if (isAlt)
401  x += size.x / 2;
402  y -= size.y / 2;
403  }
404  }

448  public void TrimAlpha()
449  {
450  var texture = GetTextureToSlice();
451  if (texture == null)
452  return;
453 
454  Rect rect = selected.rect;
455 
456  int xMin = (int)rect.xMax;
457  int xMax = (int)rect.xMin;
458  int yMin = (int)rect.yMax;
459  int yMax = (int)rect.yMin;
460 
461  for (int y = (int)rect.yMin; y < (int)rect.yMax; y++)
462  {
463  for (int x = (int)rect.xMin; x < (int)rect.xMax; x++)
464  {
465  if (PixelHasAlpha(x, y, texture))
466  {
467  xMin = Mathf.Min(xMin, x);
468  xMax = Mathf.Max(xMax, x);
469  yMin = Mathf.Min(yMin, y);
470  yMax = Mathf.Max(yMax, y);
471  }
472  }
473  }
474  // Case 582309: Return an empty rectangle if no pixel has an alpha
475  if (xMin > xMax || yMin > yMax)
476  rect = new Rect(0, 0, 0, 0);
477  else
478  rect = new Rect(xMin, yMin, xMax - xMin + 1, yMax - yMin + 1);
479 
480  if (rect.width <= 0 && rect.height <= 0)
481  {
482  m_RectsCache.Remove(selected);
483  spriteEditor.SetDataModified();
484  selected = null;
485  }
486  else
487  {
488  rect = ClampSpriteRect(rect, texture.width, texture.height);
