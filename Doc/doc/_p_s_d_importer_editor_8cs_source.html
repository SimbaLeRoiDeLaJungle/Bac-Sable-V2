588  private static bool IsAnyTextureObjectUsingPerAxisWrapMode(UnityEngine.Object[] objects, bool isVolumeTexture)
589  {
590  foreach (var o in objects)
591  {
592  int u = 0, v = 0, w = 0;
593  // the objects can be Textures themselves, or texture-related importers
594  if (o is Texture)
595  {
596  var ti = (Texture)o;
597  u = (int)ti.wrapModeU;
598  v = (int)ti.wrapModeV;
599  w = (int)ti.wrapModeW;
600  }
601  if (o is TextureImporter)
602  {
603  var ti = (TextureImporter)o;
604  u = (int)ti.wrapModeU;
605  v = (int)ti.wrapModeV;
606  w = (int)ti.wrapModeW;
607  }
608  if (o is IHVImageFormatImporter)
609  {
610  var ti = (IHVImageFormatImporter)o;
611  u = (int)ti.wrapModeU;
612  v = (int)ti.wrapModeV;
613  w = (int)ti.wrapModeW;
614  }
615  u = Mathf.Max(0, u);
616  v = Mathf.Max(0, v);
617  w = Mathf.Max(0, w);
618  if (u != v)
619  {
620  return true;
621  }
622  if (isVolumeTexture)
623  {
624  if (u != w || v != w)
625  {
626  return true;
627  }
628  }
629  }
630  return false;
631  }

638  static void WrapModePopup(SerializedProperty wrapU, SerializedProperty wrapV, SerializedProperty wrapW, bool isVolumeTexture, ref bool showPerAxisWrapModes)

644  // that seems to indicate "use defaults, user has not changed them to anything" but not totally sure.
645  // Show them as Repeat wrap modes in the popups.
646  var wu = (TextureWrapMode)Mathf.Max(wrapU.intValue, 0);
647  var wv = (TextureWrapMode)Mathf.Max(wrapV.intValue, 0);
648  var ww = (TextureWrapMode)Mathf.Max(wrapW.intValue, 0);

656  showPerAxisWrapModes = true;
657  }

659  // It's not possible to determine whether any single texture in the whole selection is using per-axis wrap modes
660  // just from SerializedProperty values. They can only tell if "some values in whole selection are different" (e.g.
661  // wrap value on U axis is not the same among all textures), and can return value of "some" object in the selection
662  // (typically based on object loading order). So in order for more intuitive behavior with multi-selection,
663  // we go over the actual objects when there's >1 object selected and some wrap modes are different.
664  if (!showPerAxisWrapModes)
665  {
666  if (wrapU.hasMultipleDifferentValues || wrapV.hasMultipleDifferentValues || (isVolumeTexture && wrapW.hasMultipleDifferentValues))

670  showPerAxisWrapModes = true;
671  }

685  wrapV.intValue = value;
686  wrapW.intValue = value;
687  showPerAxisWrapModes = false;
688  }

691  if (value == -1)
692  {
693  showPerAxisWrapModes = true;
694  EditorGUI.indentLevel++;

696  WrapModeAxisPopup(s_Styles.wrapV, wrapV);

701  EditorGUI.indentLevel--;
702  }
703  EditorGUI.showMixedValue = false;
