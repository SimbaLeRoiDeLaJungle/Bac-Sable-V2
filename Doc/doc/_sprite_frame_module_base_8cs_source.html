113  public bool Add(SpriteRect spriteRect, bool shouldReplaceInTable = false)
114  {
115  if (spriteRect.spriteID.Empty())
116  {
117  spriteRect.spriteID = GUID.Generate();
118  }
119  else
120  {
121  if (IsInternalIdInUsed(spriteRect.spriteID))
122  return false;

124 
125  if (shouldReplaceInTable)
126  {
127  // replace id from sprite to file id table
128  if (!UpdateIdInNameIdPair(spriteRect.name, spriteRect.spriteID))

134  else

136  // Since we are not replacing the file id table,

144  }

147  }

598  protected static Rect ClampSpriteRect(Rect rect, float maxX, float maxY)
599  {
600  // Clamp rect to width height
601  rect = FlipNegativeRect(rect);
602  Rect newRect = new Rect();
603

605  newRect.yMin = Mathf.Clamp(rect.yMin, 0, maxY - 1);
606  newRect.xMax = Mathf.Clamp(rect.xMax, 1, maxX);
607  newRect.yMax = Mathf.Clamp(rect.yMax, 1, maxY);
608 
609  // Prevent width and height to be 0 value after clamping.
610  if (Mathf.RoundToInt(newRect.width) == 0)
611  newRect.width = 1;
612  if (Mathf.RoundToInt(newRect.height) == 0)
613  newRect.height = 1;
