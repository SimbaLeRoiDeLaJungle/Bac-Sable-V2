30  public static LayerInfo Load(PsdBinaryReader reader, PsdFile psdFile,
31  bool globalLayerInfo, long fileEndPos)
32  {
33  Util.DebugMessage(reader.BaseStream, "Load, Begin, LayerInfo");
34 
35  // Some keys use a signature of 8B64, but the identity of these keys
36  // is undocumented. We will therefore accept either signature.
37  var signature = reader.ReadAsciiChars(4);
38  if ((signature != "8BIM") && (signature != "8B64"))
39  {
40  throw new PsdInvalidException(
41  "LayerInfo signature invalid, must be 8BIM or 8B64.");
42  }
43 
44  var key = reader.ReadAsciiChars(4);
45  var hasLongLength = LayerInfoUtil.HasLongLength(key, psdFile.IsLargeDocument);
46  LayerInfo result = new RawLayerInfo("dummy");
47  bool breakFromLoop = false;
48  while (!breakFromLoop)
49  {
50  var baseStartPosition = reader.BaseStream.Position;
51  var length = hasLongLength
52  ? reader.ReadInt64()
53  : reader.ReadInt32();
54  var startPosition = reader.BaseStream.Position;
55 
56 
57  switch (key)
58  {
59  case "Layr":
60  case "Lr16":
61  case "Lr32":
62  result = new InfoLayers(reader, psdFile, key, length);
63  break;
64  case "lsct":
65  case "lsdk":
66  result = new LayerSectionInfo(reader, key, (int)length);
67  break;
68  case "luni":
69  result = new LayerUnicodeName(reader);
70  break;
71  case "lyid":
72  result = new LayerId(reader, key, length);
73  break;
74  default:
75  result = new RawLayerInfo(reader, signature, key, length);
76  break;
77  }
78 
79  // May have additional padding applied.
80  var endPosition = startPosition + length;
81  if (reader.BaseStream.Position < endPosition)
82  reader.BaseStream.Position = endPosition;
83 
84  // Documentation states that the length is even-padded. Actually:
85  // 1. Most keys have 4-padded lengths.
86  // 2. However, some keys (LMsk) have even-padded lengths.
87  // 3. Other keys (Txt2, Lr16, Lr32) have unpadded lengths.
88  //
89  // Photoshop writes data that is always 4-padded, even when the stated
90  // length is not a multiple of 4. The length mismatch seems to occur
91  // only on global layer info. We do not read extra padding in other
92  // cases because third-party programs are likely to follow the spec.
93 
94  if (globalLayerInfo)
95  {
96  reader.ReadPadding(startPosition, 4);
97  }
98 
99  //try if we can read the next signature
100  if (reader.BaseStream.Position < fileEndPos)
101  {
102  var nowPosition = reader.BaseStream.Position;
103  signature = reader.ReadAsciiChars(4);
104  if ((signature != "8BIM") && (signature != "8B64"))
105  {
106  hasLongLength = true;
107  reader.BaseStream.Position = baseStartPosition;
108  }
109  else
110  {
111  reader.BaseStream.Position = nowPosition;
112  breakFromLoop = true;
113  }
114  }
115  else
116  breakFromLoop = true;
117  }
118 
119 
120  Util.DebugMessage(reader.BaseStream, "Load, End, LayerInfo, {0}, {1}",
121  result.Signature, result.Key);
122  return result;
123  }
124  }
