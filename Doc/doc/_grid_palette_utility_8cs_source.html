205  Sprite[] sprites = null;
206  var maxSize = Vector2.negativeInfinity;
207  var minSize = Vector2.positiveInfinity;
208 
209  // Get minimum and maximum sizes for Sprites
210  foreach (var tilemap in tilemaps)
211  {
212  var spriteCount = tilemap.GetUsedSpritesCount();
213  if (sprites == null || sprites.Length < spriteCount)
214  sprites = new Sprite[spriteCount];
215  tilemap.GetUsedSpritesNonAlloc(sprites);
216  for (int i = 0; i < spriteCount; ++i)
217  {
218  Sprite sprite = sprites[i];
219  if (sprite != null)
220  {
221  var cellSize = new Vector3(sprite.rect.width, sprite.rect.height, 0f) / sprite.pixelsPerUnit;
222  if (tilemap.cellSwizzle == GridLayout.CellSwizzle.YXZ)
223  {
224  var swap = cellSize.x;
225  cellSize.x = cellSize.y;
226  cellSize.y = swap;
227  }
228  minSize.x = Mathf.Min(cellSize.x, minSize.x);
229  minSize.y = Mathf.Min(cellSize.y, minSize.y);
230  maxSize.x = Mathf.Max(cellSize.x, maxSize.x);
231  maxSize.y = Mathf.Max(cellSize.y, maxSize.y);
232  }
233  }
234  }
235  // Validate that Sprites are in multiples of sizes
236  foreach (var tilemap in tilemaps)
237  {
238  var spriteCount = tilemap.GetUsedSpritesCount();
239  if (sprites == null || sprites.Length < spriteCount)
240  sprites = new Sprite[spriteCount];
241  tilemap.GetUsedSpritesNonAlloc(sprites);
242  for (int i = 0; i < spriteCount; ++i)
243  {
244  Sprite sprite = sprites[i];
245  if (sprite != null)
246  {
247  var cellSize = new Vector3(sprite.rect.width, sprite.rect.height, 0f) / sprite.pixelsPerUnit;
248  if (tilemap.cellSwizzle == GridLayout.CellSwizzle.YXZ)
249  {
250  var swap = cellSize.x;
251  cellSize.x = cellSize.y;
252  cellSize.y = swap;
253  }
254  // Return maximum size if sprites are not multiples of the smallest size
255  if (cellSize.x % minSize.x > 0)
256  return maxSize.x * maxSize.y <= 0f ? defaultValue : new Vector3(maxSize.x, maxSize.y, 0f);
257  if (cellSize.y % minSize.y > 0)

259  }
260  }
261  }
