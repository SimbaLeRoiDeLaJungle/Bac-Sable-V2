72  static string[] GetBurstCommandLineArgs(
73  BurstCompiler compiler,
74  NPath outputPrefixForObjectFile,
75  NPath outputDirForPatchedAssemblies,
76  string pinvokeName,
77  DotNetAssembly[] inputAssemblies)
78  {
79  var commandLineArguments = new[]
80  {
81  $"--platform={compiler.TargetPlatform}",
82  $"--target={compiler.TargetArchitecture}",
83  $"--format={compiler.ObjectFormat}",
84  compiler.SafetyChecks ? "--safety-checks" : "",
85  $"--dump=None",
86  compiler.DisableVectors ? "--disable-vectors" : "",
87  compiler.Link ? "" : "--nolink",
88  $"--float-precision={compiler.FloatPrecision}",
89  $"--keep-intermediate-files",
90  compiler.EmitLlvmObjects?"--emit-llvm-objects":"",
91  compiler.Verbose ? "--verbose" : "",
92  $"--patch-assemblies-into={outputDirForPatchedAssemblies}",
93  $"--output={outputPrefixForObjectFile}",
94  compiler.OnlyStaticMethods ? "--only-static-methods" : "",
95  "--method-prefix=burstedmethod_",
96  $"--pinvoke-name={pinvokeName}",
97  (compiler.BurstBackend == null) ? "" : $"--backend={compiler.BurstBackend}",
98  $"--execute-method-name={compiler.ExecuteMethodName}",
99  "--debug=Full",
100  compiler.EnableDirectExternalLinking ? "--enable-direct-external-linking" : "",
101  compiler.DisableOpt ? "--disable-opt" : "",
102  $"--threads={compiler.Threads}",
103  compiler.EnableGuard ? "--enable-guard" : "",
104  !String.IsNullOrEmpty(compiler.DisableWarnings) ? $"--disable-warnings={compiler.DisableWarnings}" : ""
105  }.Concat(inputAssemblies.Select(asm => $"--root-assembly={asm.Path}"));
106  if (!compiler.UseOwnToolchain)
107  commandLineArguments = commandLineArguments.Concat(new[] {"--no-native-toolchain"});
108 
109 
110  if (!HostPlatform.IsWindows)
111  commandLineArguments = new[] {BurstExecutable.ToString(SlashMode.Native)}.Concat(commandLineArguments);
112  if (compiler.EnableStaticLinkage)
113  commandLineArguments = commandLineArguments.Concat(new[] {"--generate-static-linkage-methods"});
114  if (compiler.EnableJobMarshalling)
115  commandLineArguments = commandLineArguments.Concat(new[] { "--generate-job-marshalling-methods" });
116 
117  return commandLineArguments.ToArray();
118  }

132  static string[] ArgumentsToResponseFile(bool usesMono, NPath targetDir, string[] commandLineArgs, out NPath responseFile)
133  {
134  var remainingArgs = usesMono? commandLineArgs.Skip(1) : commandLineArgs;
135  var joinedArgs = String.Join(Environment.NewLine, remainingArgs.Where(s => !String.IsNullOrEmpty(s)));
136  KnuthHash hash = new KnuthHash();
137  hash.Add(joinedArgs);
138  responseFile = targetDir.Combine($"{hash.Value}.rsp");
139 
140  Backend.Current.AddWriteTextAction(
141  responseFile,
142  joinedArgs,
143  "BurstResponseFile");
144 
145  if (usesMono)
146  {
147  return new [] { commandLineArgs[0], $"@{responseFile}"};
148  }
149  return new [] {$"@{responseFile}"};
150  }

152  public static BagOfObjectFilesLibrary SetupBurstCompilationForAssemblies(
153  BurstCompiler compiler,
154  DotNetAssembly unpatchedInputAssembly,
155  NPath outputDirForObjectFile,
156  NPath outputDirForPatchedAssemblies,
157  string pinvokeName,
158  out DotNetAssembly patchedAssembly)
159  {
160  /*
161  * Note that you can have Link be true and still use this, because on iOS for example if you
162  * DON'T pass --no-link, it will NOT link, but it WILL correctly override the llvm backend.
163  *
164  * if you DO pass --no-link, it will also not link, but then incorrectly use llvm 9.
165  */
166 
167  patchedAssembly = unpatchedInputAssembly.ApplyDotNetAssembliesPostProcessor(
168  outputDirForPatchedAssemblies,
169  (inputAssemblies, targetDir) =>
170  {
171  var usesMono = HostPlatform.IsWindows ? false : true;
172  var executableStringFor = !usesMono ? BurstExecutable.ToString(SlashMode.Native) : "mono";
173  var commandLineArgs = GetBurstCommandLineArgs(
174  compiler,
175  outputDirForObjectFile.Combine(pinvokeName),
176  outputDirForPatchedAssemblies,
177  pinvokeName,
178  inputAssemblies);
179 
180  var inputPaths = AddDebugSymbolPaths(inputAssemblies);
181  var targetFiles = inputPaths.Select(p => targetDir.Combine(p.FileName));
182 
183  var finalArguments = ArgumentsToResponseFile(usesMono, targetDir, commandLineArgs, out var responseFile);
184 
185  Backend.Current.AddAction(
186  "Burst",
187  //todo: make burst process pdbs
188  targetFiles.ToArray(),
189  inputPaths.Concat(new[] {BurstExecutable, responseFile}).ToArray(),
190  executableStringFor,
191  finalArguments,
192  targetDirectories: new[] {outputDirForObjectFile}
193  );
194  });
195  var needFake = true;
196  NPath[] objectFileList = null;
197  if (outputDirForObjectFile.Exists())
198  {
199  objectFileList = outputDirForObjectFile.Files($"*{compiler.ObjectFileExtension}");
200  if (objectFileList.Length > 0)
201  needFake = false;
202  }
203  if (needFake)
204  objectFileList = new[] {outputDirForObjectFile.Combine($"fake{compiler.ObjectFileExtension}")};
205  return new BagOfObjectFilesLibrary(objectFileList);
206  }
