55  static void RemapCharacterPartsToNewBones(SkinningCache skinningCache, SpriteBone[] newBones)
56  {
57  var skeleton = skinningCache.character.skeleton;
58  var previousStateBones = skeleton.bones;
59  var skeletonBones = skinningCache.CreateBoneCacheFromSpriteBones(newBones, 1.0f);
60  skeleton.SetBones(skeletonBones);
61 
62  for (var i = 0; i < skinningCache.character.parts.Length; i++)
63  {
64  var characterPart = skinningCache.character.parts[i];
65  var useGuids = !skeletonBones.All(newBone => previousStateBones.All(oldBone => newBone.guid != oldBone.guid));
66  characterPart.bones = useGuids ?
67  characterPart.bones.Select(partBone => Array.Find(skeletonBones, skeletonBone => partBone.guid == skeletonBone.guid)).ToArray() :
68  characterPart.bones.Select(partBone => skeletonBones.ElementAtOrDefault(Array.FindIndex(previousStateBones, oldBone => partBone.guid == oldBone.guid))).ToArray();
69 
70  var mesh = skinningCache.GetMesh(characterPart.sprite);
71  if (mesh != null)
72  mesh.SetCompatibleBoneSet(characterPart.bones);
73 
74  skinningCache.character.parts[i] = characterPart;
75  }

89  static void CalculateLocaltoWorldMatrix(int i, SpriteRect spriteRect, float definitionScale, float pixelsPerUnit, List<UnityEngine.U2D.SpriteBone> spriteBone, ref UnityEngine.U2D.SpriteBone?[] outpriteBone, ref NativeArray<Matrix4x4> bindPose)
90  {
91  if (outpriteBone[i] != null)
92  return;
93  UnityEngine.U2D.SpriteBone sp = spriteBone[i];
94  var isRoot = sp.parentId == -1;
95  var position = isRoot ? (spriteBone[i].position - Vector3.Scale(spriteRect.rect.size, spriteRect.pivot)) : spriteBone[i].position;
96  position.z = 0f;
97  sp.position = position * definitionScale / pixelsPerUnit;
98  sp.length = spriteBone[i].length * definitionScale / pixelsPerUnit;
99  outpriteBone[i] = sp;
100 
101  // Calculate bind poses
102  var worldPosition = Vector3.zero;
103  var worldRotation = Quaternion.identity;
104 
105  if (sp.parentId == -1)
106  {
107  worldPosition = sp.position;
108  worldRotation = sp.rotation;
109  }
110  else
111  {
112  if (outpriteBone[sp.parentId] == null)
113  {
114  CalculateLocaltoWorldMatrix(sp.parentId, spriteRect, definitionScale, pixelsPerUnit, spriteBone, ref outpriteBone, ref bindPose);
115  }
116  var parentBindPose = bindPose[sp.parentId];
117  var invParentBindPose = Matrix4x4.Inverse(parentBindPose);
118 
119  worldPosition = invParentBindPose.MultiplyPoint(sp.position);
120  worldRotation = sp.rotation * invParentBindPose.rotation;
121  }
122 
123  // Practically Matrix4x4.SetTRInverse
124  var rot = Quaternion.Inverse(worldRotation);
125  Matrix4x4 mat = Matrix4x4.identity;
126  mat = Matrix4x4.Rotate(rot);
127  mat = mat * Matrix4x4.Translate(-worldPosition);
128 
129 
130  bindPose[i] = mat;
131  }

133  static bool PostProcessBoneData(ISpriteEditorDataProvider spriteDataProvider, float definitionScale, Sprite[] sprites)
134  {
135  var boneDataProvider = spriteDataProvider.GetDataProvider<ISpriteBoneDataProvider>();
136  var textureDataProvider = spriteDataProvider.GetDataProvider<ITextureDataProvider>();
137 
138  if (sprites == null || sprites.Length == 0 || boneDataProvider == null || textureDataProvider == null)
139  return false;
140 
141  var dataChanged = false;
142  var spriteRects = spriteDataProvider.GetSpriteRects();
143  foreach (var sprite in sprites)
144  {
145  var guid = sprite.GetSpriteID();
146  {
147  var spriteBone = boneDataProvider.GetBones(guid);
148  if (spriteBone == null)
149  continue;
150 
151  var spriteBoneCount = spriteBone.Count;
152  if (spriteBoneCount == 0)
153  continue;
154 
155  var spriteRect = spriteRects.First(s => { return s.spriteID == guid; });
156 
157  var bindPose = new NativeArray<Matrix4x4>(spriteBoneCount, Allocator.Temp);
158  var outputSpriteBones = new UnityEngine.U2D.SpriteBone ? [spriteBoneCount];
159  for (int i = 0; i < spriteBoneCount; ++i)
160  {
161  CalculateLocaltoWorldMatrix(i, spriteRect, definitionScale, sprite.pixelsPerUnit, spriteBone, ref outputSpriteBones, ref bindPose);
162  }
163  sprite.SetBindPoses(bindPose);
164  sprite.SetBones(outputSpriteBones.Select(x => x.Value).ToArray());
165  bindPose.Dispose();
166 
167  dataChanged = true;
168  }
169  }
170 
171  return dataChanged;
172  }

174  static bool PostProcessSpriteMeshData(ISpriteEditorDataProvider spriteDataProvider, float definitionScale, Sprite[] sprites)
175  {
176  var spriteMeshDataProvider = spriteDataProvider.GetDataProvider<ISpriteMeshDataProvider>();
177  var boneDataProvider = spriteDataProvider.GetDataProvider<ISpriteBoneDataProvider>();
178  var textureDataProvider = spriteDataProvider.GetDataProvider<ITextureDataProvider>();
179  if (sprites == null || sprites.Length == 0 || spriteMeshDataProvider == null || textureDataProvider == null)
180  return false;
181 
182  var dataChanged = false;
183  var spriteRects = spriteDataProvider.GetSpriteRects();
184  foreach (var sprite in sprites)
185  {
186  var guid = sprite.GetSpriteID();
187  var vertices = spriteMeshDataProvider.GetVertices(guid);
188  int[] indices = null;
189  if (vertices.Length > 2)
190  indices = spriteMeshDataProvider.GetIndices(guid);
191 
192  if (indices != null && indices.Length > 2 && vertices.Length > 2 )
193  {
194  var spriteRect = spriteRects.First(s => { return s.spriteID == guid; });
195  var spriteBone = boneDataProvider.GetBones(guid);
196  var hasBones = spriteBone != null && spriteBone.Count > 0;
197  var hasInvalidWeights = false;
198 
199  var vertexArray = new NativeArray<Vector3>(vertices.Length, Allocator.Temp);
200  var boneWeightArray = new NativeArray<BoneWeight>(vertices.Length, Allocator.Temp);
201 
202  for (int i = 0; i < vertices.Length; ++i)
203  {
204  var boneWeight = vertices[i].boneWeight;
205 
206  vertexArray[i] = (Vector3)(vertices[i].position - Vector2.Scale(spriteRect.rect.size, spriteRect.pivot)) * definitionScale / sprite.pixelsPerUnit;
207  boneWeightArray[i] = boneWeight;
208 
209  if (hasBones && !hasInvalidWeights)
210  {
211  var sum = boneWeight.weight0 + boneWeight.weight1 + boneWeight.weight2 + boneWeight.weight3;
212  hasInvalidWeights = sum < 0.999f;
213  }
214  }
215 
216  var indicesArray = new NativeArray<ushort>(indices.Length, Allocator.Temp);
217  for (int i = 0; i < indices.Length; ++i)
218  indicesArray[i] = (ushort)indices[i];
219 
220  sprite.SetVertexCount(vertices.Length);
221  sprite.SetVertexAttribute<Vector3>(VertexAttribute.Position, vertexArray);
222  sprite.SetIndices(indicesArray);
223  sprite.SetVertexAttribute<BoneWeight>(VertexAttribute.BlendWeight, boneWeightArray);
224  vertexArray.Dispose();
225  boneWeightArray.Dispose();
226  indicesArray.Dispose();
227 
228  // Deformed Sprites require proper Tangent Channels if Lit. Enable Tangent channels.
229  if (hasBones)
230  {
231  var tangentArray = new NativeArray<Vector4>(vertices.Length, Allocator.Temp);
232  for (int i = 0; i < vertices.Length; ++i)
233  tangentArray[i] = new Vector4(1.0f, 0.0f, 0, -1.0f);
234  sprite.SetVertexAttribute<Vector4>(VertexAttribute.Tangent, tangentArray);
235  tangentArray.Dispose();
236  }
237 
238  dataChanged = true;
239 
240  if (hasBones && hasInvalidWeights)
241  Debug.LogWarning("Sprite \"" + spriteRect.name + "\" contains bone weights which sum zero or are not normalized. To avoid visual artifacts please consider fixing them.");
242  }
243  else
244  {
245  var boneWeightArray = new NativeArray<BoneWeight>(sprite.GetVertexCount(), Allocator.Temp);
246  var defaultBoneWeight = new BoneWeight() { weight0 = 1f };
247 
248  for (var i = 0; i < boneWeightArray.Length; ++i)
249  boneWeightArray[i] = defaultBoneWeight;
250 
251  sprite.SetVertexAttribute<BoneWeight>(VertexAttribute.BlendWeight, boneWeightArray);
252  }
253  }
254 
255  return dataChanged;

258  static float CalculateDefinitionScale(Texture2D texture, ITextureDataProvider dataProvider)
259  {
260  float definitionScale = 1;
261  if (texture != null && dataProvider != null)
262  {
263  int actualWidth = 0, actualHeight = 0;
264  dataProvider.GetTextureActualWidthAndHeight(out actualWidth, out actualHeight);
265  float definitionScaleW = texture.width / (float)actualWidth;

267  definitionScale = Mathf.Min(definitionScaleW, definitionScaleH);
