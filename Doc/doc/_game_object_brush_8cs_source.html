222  public override void Rotate(RotationDirection direction, GridLayout.CellLayout layout)
223  {
224  Vector3Int oldSize = m_Size;
225  BrushCell[] oldCells = m_Cells.Clone() as BrushCell[];
226  size = new Vector3Int(oldSize.y, oldSize.x, oldSize.z);
227  BoundsInt oldBounds = new BoundsInt(Vector3Int.zero, oldSize);
228 
229  foreach (Vector3Int oldPos in oldBounds.allPositionsWithin)
230  {
231  int newX = direction == RotationDirection.Clockwise ? oldSize.y - oldPos.y - 1 : oldPos.y;
232  int newY = direction == RotationDirection.Clockwise ? oldPos.x : oldSize.x - oldPos.x - 1;
233  int toIndex = GetCellIndex(newX, newY, oldPos.z);
234  int fromIndex = GetCellIndex(oldPos.x, oldPos.y, oldPos.z, oldSize.x, oldSize.y, oldSize.z);
235  m_Cells[toIndex] = oldCells[fromIndex];
236  }

238  int newPivotX = direction == RotationDirection.Clockwise ? oldSize.y - pivot.y - 1 : pivot.y;
239  int newPivotY = direction == RotationDirection.Clockwise ? pivot.x : oldSize.x - pivot.x - 1;
240  pivot = new Vector3Int(newPivotX, newPivotY, pivot.z);
241 
242  Matrix4x4 rotation = Matrix4x4.TRS(Vector3.zero, Quaternion.Euler(0f, 0f, direction == RotationDirection.Clockwise ? 90f : -90f), Vector3.one);
243  Quaternion orientation = Quaternion.Euler(0f, 0f, direction == RotationDirection.Clockwise ? 90f : -90f);

519  private GameObject GetObjectInCell(GridLayout grid, Transform parent, Vector3Int position)
520  {
521  int childCount;
522  GameObject[] sceneChildren = null;
523  if (parent == null)
524  {
525  var scene = SceneManager.GetActiveScene();
526  sceneChildren = scene.GetRootGameObjects();
527  childCount = scene.rootCount;
528  }
529  else
530  {
531  childCount = parent.childCount;
532  }
533  var anchorCellOffset = Vector3Int.FloorToInt(m_Anchor);
534  var cellSize = grid.cellSize;
535  anchorCellOffset.x = cellSize.x == 0 ? 0 : anchorCellOffset.x;
536  anchorCellOffset.y = cellSize.y == 0 ? 0 : anchorCellOffset.y;
537  anchorCellOffset.z = cellSize.z == 0 ? 0 : anchorCellOffset.z;
538 
539  for (var i = 0; i < childCount; i++)
540  {
541  var child = sceneChildren == null ? parent.GetChild(i) : sceneChildren[i].transform;
542  if (position == grid.WorldToCell(child.position) - anchorCellOffset)
543  return child.gameObject;
544  }
545  return null;
546  }
