352  Quaternion gameObjectRotation = new Quaternion();

354 
355  bool ruleMatched = false;
356  Matrix4x4 transform = iden;
357  foreach (TilingRule rule in m_TilingRules)
358  {
359  if (RuleMatches(rule, position, tilemap, ref transform))
360  {
361  transform = orientMatrix * transform;
362 
363  // Converts the tile's translation, rotation, & scale matrix to values to be used by the instantiated GameObject
364  gameObjectTranslation = new Vector3(transform.m03, transform.m13, transform.m23);
365  gameObjectRotation = Quaternion.LookRotation(new Vector3(transform.m02, transform.m12, transform.m22), new Vector3(transform.m01, transform.m11, transform.m21));
366  gameObjectScale = transform.lossyScale;
367 
368  ruleMatched = true;
369  break;
370  }
371  }
372  if (!ruleMatched)
373  {
374  // Fallback to just using the orientMatrix for the translation, rotation, & scale values.
375  gameObjectTranslation = new Vector3(orientMatrix.m03, orientMatrix.m13, orientMatrix.m23);
376  gameObjectRotation = Quaternion.LookRotation(new Vector3(orientMatrix.m02, orientMatrix.m12, orientMatrix.m22), new Vector3(orientMatrix.m01, orientMatrix.m11, orientMatrix.m21));
377  gameObjectScale = orientMatrix.lossyScale;
378  }
379 
380  instantiatedGameObject.transform.localPosition = gameObjectTranslation + tmpMap.CellToLocalInterpolated(position + tmpMap.tileAnchor);
381  instantiatedGameObject.transform.localRotation = gameObjectRotation;
