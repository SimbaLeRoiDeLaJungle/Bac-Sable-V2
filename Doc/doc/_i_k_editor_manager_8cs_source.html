187  {

189  return;
190  foreach (var solver in manager.solvers)
191  {
192  if (solver == null || !solver.isActiveAndEnabled)
193  continue;
194 
195  if (!solver.isValid)
196  solver.Initialize();
197 
198  if (!solver.isValid)
199  continue;
200 
201  for (int i = 0; i < solver.chainCount; ++i)
202  {

204

213  }
214 
215

217  Undo.RecordObject(chain.target, undoName);

222  {
223  if(m_IgnoreTransformsOnUndo.Contains(t))
224  continue;

227  }
228

230  Undo.RegisterCompleteObjectUndo(chain.target, undoName);
