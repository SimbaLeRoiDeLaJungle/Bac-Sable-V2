183  static bool WithinRange(ShapeAngleRange angleRange, float inputAngle)
184  {
185  float range = angleRange.end - angleRange.start;
186  float angle = Mathf.Repeat(inputAngle - angleRange.start, 360f);
187  angle = (angle == 360.0f) ? 0 : angle;
188  return (angle >= 0f && angle <= range);
189  }

203  {
204  List<ShapeAngleRange> angleRanges = new List<ShapeAngleRange>();

216  private void GenerateSegments(SpriteShapeController sc, List<ShapeAngleRange> angleRanges)
217  {
218  var controlPointCount = sc.spline.GetPointCount();
219  var angleRangeIndices = new int[controlPointCount];
220  ShapeSegment activeSegment = new ShapeSegment() { start = -1, end = -1, angleRange = -1 };
221  m_ShapeSegments.Clear();
222 
223  for (int i = 0; i < controlPointCount; ++i)
224  {
225  var actv = i;
226  var next = SplineUtility.NextIndex(actv, controlPointCount);
227  var pos1 = sc.spline.GetPosition(actv);
228  var pos2 = sc.spline.GetPosition(next);
229  bool continueStrip = (sc.spline.GetTangentMode(actv) == ShapeTangentMode.Continuous), edgeUpdated = false;
230  float angle = 0;
231  if (false == continueStrip || activeSegment.start == -1)
232  angle = SplineUtility.SlopeAngle(pos1, pos2) + 90.0f;
233 
234  next = (!sc.spline.isOpenEnded && next == 0) ? (actv + 1) : next;
235  int mn = (actv < next) ? actv : next;
236  int mx = (actv > next) ? actv : next;
237 
238  var anglerange = RangeFromAngle(angleRanges, angle);
239  angleRangeIndices[actv] = anglerange;
240  if (anglerange == -1)
241  {
242  activeSegment = new ShapeSegment() { start = mn, end = mx, angleRange = anglerange };
243  m_ShapeSegments.Add(activeSegment);
244  continue;
245  }
246 
247  // Check for Segments. Also check if the Segment Start has been resolved. Otherwise simply start with the next one
248  if (activeSegment.start != -1)
249  continueStrip = continueStrip && (angleRangeIndices[activeSegment.start] != -1);
250 
251  bool canContinue = (actv != (controlPointCount - 1)) || (!sc.spline.isOpenEnded && (actv == (controlPointCount - 1)));
252  if (continueStrip && canContinue)
253  {
254  for (int s = 0; s < m_ShapeSegments.Count; ++s)
255  {
256  activeSegment = m_ShapeSegments[s];
257  if (activeSegment.start - mn == 1)
258  {
259  edgeUpdated = true;
260  activeSegment.start = mn;
261  m_ShapeSegments[s] = activeSegment;
262  break;
263  }
264  if (mx - activeSegment.end == 1)
265  {
266  edgeUpdated = true;
267  activeSegment.end = mx;
268  m_ShapeSegments[s] = activeSegment;
269  break;
270  }
271  }
272  }
273 
274  if (!edgeUpdated)
275  {
276  activeSegment.start = mn;
277  activeSegment.end = mx;
278  activeSegment.angleRange = anglerange;
279  m_ShapeSegments.Add(activeSegment);
280  }
281 
282  }
283  }

318  private int ResolveSpriteIndex(List<int> spriteIndices, ISelection<int> selection, ref List<int> startPoints)
319  {
320  var spriteIndexValue = spriteIndices.FirstOrDefault();
321  var sc = target as SpriteShapeController;
322  var spline = sc.spline;
323 
324  if (sc == null || sc.spriteShape == null)
325  return -1;
326  UpdateSegments();
327 
328  if (sc.spriteShape != null)
329  {
330  if (selection.Count == 1)
331  {
332  m_SelectedAngleRange = GetAngleRange(sc, selection.elements[0], ref m_SelectedPoint);
333  startPoints.Add(m_SelectedPoint);
334  spriteIndexValue = spline.GetSpriteIndex(m_SelectedPoint);
335  }
336  else
337  {
338  m_SelectedAngleRange = -1;
339  foreach (var index in selection.elements)
340  {
341  int startPoint = index;
342  int angleRange = GetAngleRange(sc, index, ref startPoint);
343  if (m_SelectedAngleRange != -1 && angleRange != m_SelectedAngleRange)
344  {
345  m_SelectedAngleRange = -1;
346  break;
347  }
348  startPoints.Add(startPoint);
349  m_SelectedAngleRange = angleRange;
350  }
351  }
352  }
353 
354  if (m_SelectedAngleRange != -1)
355  spriteSelector.UpdateSprites(sc.spriteShape.angleRanges[m_SelectedAngleRange].sprites.ToArray());
356  else
357  spriteIndexValue = -1;
358  return spriteIndexValue;
359  }
