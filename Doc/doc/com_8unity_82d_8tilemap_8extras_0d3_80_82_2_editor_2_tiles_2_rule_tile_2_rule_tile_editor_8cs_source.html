240  private void UpdateTilingRuleIds()
241  {

243  var usedIdSet = new HashSet<int>();
244  foreach (var rule in tile.m_TilingRules)

372  rule.m_Sprites[0] = tile.m_DefaultSprite;

375 
376  var count = m_TilingRules.arraySize;

378

709  {
710  if (Event.current.type == EventType.MouseDown && ContainsMousePosition(rect))
711  {
712  var allConsts = tile.m_NeighborType.GetFields(BindingFlags.Public | BindingFlags.Static | BindingFlags.FlattenHierarchy);

714  neighborConsts.Sort();
715 
716  if (neighbors.ContainsKey(position))
717  {
718  int oldIndex = neighborConsts.IndexOf(neighbors[position]);
719  int newIndex = oldIndex + GetMouseChange();
720  if (newIndex >= 0 && newIndex < neighborConsts.Count)
721  {

729  }

731  {
