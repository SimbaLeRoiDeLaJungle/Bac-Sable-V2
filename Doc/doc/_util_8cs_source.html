230  public static int RoundUp(int value, int multiple)
231  {

373  public UFixed16_16(double value)
374  {
375  if (value >= 65536.0) throw new OverflowException();
376  if (value < 0) throw new OverflowException();
377 
378  Integer = (UInt16)value;
379 
380  // Round instead of truncate, because doubles may not represent the
381  // fraction exactly.
382  Fraction = (UInt16)((value - Integer) * 65536 + 0.5);
383  }
