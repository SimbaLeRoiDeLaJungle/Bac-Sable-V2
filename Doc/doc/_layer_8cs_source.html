141  public Layer(PsdBinaryReader reader, PsdFile psdFile)
142  : this(psdFile)
143  {
144  Util.DebugMessage(reader.BaseStream, "Load, Begin, Layer");
145 
146  Rect = reader.ReadRectangle();
147 
148  //-----------------------------------------------------------------------
149  // Read channel headers. Image data comes later, after the layer header.
150 
151  int numberOfChannels = reader.ReadUInt16();
152  for (int channel = 0; channel < numberOfChannels; channel++)
153  {
154  var ch = new Channel(reader, this);
155  Channels.Add(ch);
156  }
157 
158  //-----------------------------------------------------------------------
159  //
160 
161  var signature = reader.ReadAsciiChars(4);
162  if (signature != "8BIM")
163  throw (new PsdInvalidException("Invalid signature in layer header."));
164 
165  BlendModeKey = reader.ReadAsciiChars(4);
166  Opacity = reader.ReadByte();
167  Clipping = reader.ReadBoolean();
168 
169  var flagsByte = reader.ReadByte();
170  flags = new BitVector32(flagsByte);
171  reader.ReadByte(); //padding
172 
173  //-----------------------------------------------------------------------
174 
175  // This is the total size of the MaskData, the BlendingRangesData, the
176  // Name and the AdjustmentLayerInfo.
177  var extraDataSize = reader.ReadUInt32();
178  var extraDataStartPosition = reader.BaseStream.Position;
179 
180  Masks = new MaskInfo(reader, this);
181  BlendingRangesData = new BlendingRanges(reader, this);
182  Name = reader.ReadPascalString(4);
183 
184  //-----------------------------------------------------------------------
185  // Process Additional Layer Information
186 
187  long adjustmentLayerEndPos = extraDataStartPosition + extraDataSize;
188  while (reader.BaseStream.Position < adjustmentLayerEndPos)
189  {
190  var layerInfo = LayerInfoFactory.Load(reader, this.PsdFile, false, adjustmentLayerEndPos);
191  AdditionalInfo.Add(layerInfo);
192  }
193 
194  foreach (var adjustmentInfo in AdditionalInfo)
195  {
196  switch (adjustmentInfo.Key)
197  {
198  case "luni":
199  Name = ((LayerUnicodeName)adjustmentInfo).Name;
200  break;
201  case "lyid":
202  LayerID = ((LayerId)adjustmentInfo).ID;
203  break;
204  }
205  }
206 
207  Util.DebugMessage(reader.BaseStream, "Load, End, Layer, {0}", Name);
208 
209  PsdFile.LoadContext.OnLoadLayerHeader(this);
210  }

217  public void CreateMissingChannels()
218  {
219  var channelCount = this.PsdFile.ColorMode.MinChannelCount();
220  for (short id = 0; id < channelCount; id++)
221  {
222  if (!this.Channels.ContainsId(id))
223  {
224  var size = this.Rect.Height * this.Rect.Width;
225 
226  var ch = new Channel(id, this);
227  ch.ImageData = new NativeArray<byte>(size, Allocator.TempJob);
228  unsafe
229  {
230  UnsafeUtility.MemSet(ch.ImageData.GetUnsafePtr(), (byte)255, size);
231  }
232  this.Channels.Add(ch);
233  }
234  }
