122  public MaskInfo(PsdBinaryReader reader, Layer layer)
123  {
124  Util.DebugMessage(reader.BaseStream, "Load, Begin, MaskInfo");
125 
126  var maskLength = reader.ReadUInt32();
127  if (maskLength <= 0)
128  return;
129 
130  var startPosition = reader.BaseStream.Position;
131  var endPosition = startPosition + maskLength;
132 
133  // Read layer mask
134  var rectangle = reader.ReadRectangle();
135  var backgroundColor = reader.ReadByte();
136  var flagsByte = reader.ReadByte();
137  LayerMask = new Mask(layer, rectangle, backgroundColor, new BitVector32(flagsByte));
138 
139  // User mask is supplied separately when there is also a vector mask.
140  if (maskLength == 36)
141  {
142  var userFlagsByte = reader.ReadByte();
143  var userBackgroundColor = reader.ReadByte();
144  var userRectangle = reader.ReadRectangle();
145  UserMask = new Mask(layer, userRectangle, userBackgroundColor, new BitVector32(userFlagsByte));
146  }
147  else
148  {
149  // Only check if bit 4 is set. Testing shows there are discrepancy in file format documentation.
150  if (flagsByte == 16)
151  {
152  // Not using them so just read and discard the values
153  var maskParameters = new BitVector32(reader.ReadByte());
154  if (maskParameters[s_UserMaskDensityBit])
155  reader.ReadByte();
156 
157  if (maskParameters[s_UserMaskFeatherBit])
158  reader.ReadDouble();
159 
160  if (maskParameters[s_VectorMaskDensityBit])
161  reader.ReadByte();
162 
163  if (maskParameters[s_VectorMaskFeatherBit])
164  reader.ReadDouble();
165  }
166 
167  // The rest should be vector mask
168  if (reader.BaseStream.Position + 18 <= endPosition)
169  {
170  var userFlagsByte = reader.ReadByte();
171  var userBackgroundColor = reader.ReadByte();
172  var userRectangle = reader.ReadRectangle();
173  UserMask = new Mask(layer, userRectangle, userBackgroundColor, new BitVector32(userFlagsByte));
174  }
175  }
176 
177  // 20-byte mask data will end with padding.
178  reader.BaseStream.Position = endPosition;
179 
180  Util.DebugMessage(reader.BaseStream, "Load, End, MaskInfo");
181  }
182  }
