588  if (paletteAsset != null)

591  {

593  if (grid != null)

596  }

600  }
601  }

605  }

1018  private GameObject FindPrefabInstanceEquivalent(GameObject prefabInstance, GameObject prefabTarget)
1019  {
1020  var prefabRoot = prefabTarget.transform.root.gameObject;
1021  var currentTransform = prefabTarget.transform;
1022  var reverseTransformOrder = new Stack<int>();
1023  while (currentTransform != prefabRoot.transform && currentTransform.parent != null)
1024  {
1025  var parentTransform = currentTransform.parent;
1026  for (int i = 0; i < parentTransform.childCount; ++i)
1027  {
1028  if (currentTransform == parentTransform.GetChild(i))
1029  {
1030  reverseTransformOrder.Push(i);
1031  break;
1032  }
1033  }
1034  currentTransform = currentTransform.parent;
1035  }
1036 
1037  currentTransform = prefabInstance.transform;
1038  while (reverseTransformOrder.Count > 0)
1039  {
1040  var childIndex = reverseTransformOrder.Pop();
1041  if (childIndex >= currentTransform.childCount)
1042  return null;
1043  currentTransform = currentTransform.GetChild(childIndex);
1044  }
1045  return currentTransform.gameObject;
1046  }
