87  public static void Normalize(ref Vec3 v)
88  {

263  public static Edge MakeEdge(Edge eNext)
264  {
265  Debug.Assert(eNext != null);
266 
267  var pair = EdgePair.Create();
268  var e = pair._e;
269  var eSym = pair._eSym;
270 
271  // Make sure eNext points to the first edge of the edge pair
272  Edge.EnsureFirst(ref eNext);
273 
274  // Insert in circular doubly-linked list before eNext.
275  // Note that the prev pointer is stored in Sym->next.
276  var ePrev = eNext._Sym._next;
277  eSym._next = ePrev;
278  ePrev._Sym._next = e;
279  e._next = eNext;
280  eNext._Sym._next = eSym;
281 
282  e._Sym = eSym;
283  e._Onext = e;
284  e._Lnext = eSym;
285  e._Org = null;
286  e._Lface = null;
287  e._winding = 0;
288  e._activeRegion = null;
289 
290  eSym._Sym = e;
291  eSym._Onext = eSym;
292  eSym._Lnext = e;
293  eSym._Org = null;
294  eSym._Lface = null;
295  eSym._winding = 0;
296  eSym._activeRegion = null;
297 
298  return e;

326  public static void MakeVertex(Edge eOrig, Vertex vNext)
327  {

329

331  var vPrev = vNext._prev;
332  vNew._prev = vPrev;
333  vPrev._next = vNew;

335  vNext._prev = vNew;

451  public static Real FaceArea(Face f)
