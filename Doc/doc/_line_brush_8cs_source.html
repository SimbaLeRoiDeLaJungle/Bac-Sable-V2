109  {

111  if (fillGaps)
112  {
113  var rise = endPos.y - startPos.y;
114  var run = endPos.x - startPos.x;
115 
116  if (rise != 0 || run != 0)
117  {
118  var extraStart = startPos;
119  var extraEnd = endPos;
120 
121 
122  if (Mathf.Abs(rise) >= Mathf.Abs(run))
123  {
124  // up
125  if (rise > 0)
126  {
127  extraStart.y += 1;
128  extraEnd.y += 1;
129  }
130  // down
131  else // rise < 0
132  {
133 
134  extraStart.y -= 1;
135  extraEnd.y -= 1;
136  }
137  }
138  else // Mathf.Abs(rise) < Mathf.Abs(run)
139  {
140 
141  // right
142  if (run > 0)
143  {
144  extraStart.x += 1;
145  extraEnd.x += 1;
146  }
147  // left
148  else // run < 0
149  {
150  extraStart.x -= 1;
151  extraEnd.x -= 1;
152  }
153  }

156  extraPoints = extraPoints.Except(new[] { extraEnd });

172  public static IEnumerable<Vector2Int> GetPointsOnLine(Vector2Int p1, Vector2Int p2)
173  {
174  int x0 = p1.x;
175  int y0 = p1.y;
176  int x1 = p2.x;
177  int y1 = p2.y;
178 
179  bool steep = Math.Abs(y1 - y0) > Math.Abs(x1 - x0);
180  if (steep)
181  {
182  int t;
183  t = x0; // swap x0 and y0
184  x0 = y0;
185  y0 = t;
186  t = x1; // swap x1 and y1
187  x1 = y1;
188  y1 = t;
189  }
190  if (x0 > x1)
191  {
192  int t;
193  t = x0; // swap x0 and x1
194  x0 = x1;
195  x1 = t;
196  t = y0; // swap y0 and y1
197  y0 = y1;
198  y1 = t;
199  }
200  int dx = x1 - x0;
201  int dy = Math.Abs(y1 - y0);
202  int error = dx / 2;
203  int ystep = (y0 < y1) ? 1 : -1;
204  int y = y0;
205  for (int x = x0; x <= x1; x++)
206  {
207  yield return new Vector2Int((steep ? y : x), (steep ? x : y));
208  error = error - dy;
209  if (error < 0)
210  {
211  y += ystep;
212  error += dx;
213  }
214  }
215  yield break;
216  }
