59  private void Predict(/*UInt16**/ byte[] ptrData)
60  {
61  int size = sizeof(UInt16);
62  // Delta-encode each row
63  for (int i = 0; i < Size.Height; i++)
64  {
65  int rowOffset = Size.Width * i * size;
66  //UInt16* ptrDataRow = ptrData;
67  var ptrDataRowEnd = Size.Width - 1;
68 
69  // Start with the last column in the row
70  while (ptrDataRowEnd > 0)
71  {
72  var v = BitConverter.ToUInt16(ptrData, ptrDataRowEnd * size + rowOffset);
73  var v1 = BitConverter.ToUInt16(ptrData, (ptrDataRowEnd - 1) * size + rowOffset);
74  v -= v1;
75  var b = BitConverter.GetBytes(v);
76  for (int c = 0; c < b.Length; ++c)
77  {
78  ptrData[ptrDataRowEnd * size + rowOffset + c] = b[c];
79  }
80  ptrDataRowEnd--;
81  }
82  }
83  }

88  private void Unpredict(byte[] ptrData)
89  {
90  int size = sizeof(UInt16);
91  // Delta-decode each row
92  for (int i = 0; i < Size.Height; i++)
93  {
94  //UInt16* ptrDataRowEnd = ptrData + Size.Width;
95  int rowOffset = Size.Width * i * size;
96  // Start with column index 1 on each row
97  int start = 1;
98  while (start < Size.Width)
99  {
100  var v = BitConverter.ToUInt16(ptrData, start * size + rowOffset);
101  var v1 = BitConverter.ToUInt16(ptrData, (start - 1) * size + rowOffset);
102  v += v1;
103  var b = BitConverter.GetBytes(v);
104  for (int c = 0; c < b.Length; ++c)
105  {
106  ptrData[start * size + rowOffset + c] = b[c];
107  }
108  start++;
109  }
