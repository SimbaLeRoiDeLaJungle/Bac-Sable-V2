142  internal static class ImageResourceFactory
143  {
144  public static ImageResource CreateImageResource(PsdBinaryReader reader)
145  {
146  Util.DebugMessage(reader.BaseStream, "Load, Begin, ImageResource");
147 
148  var signature = reader.ReadAsciiChars(4);
149  var resourceIdInt = reader.ReadUInt16();
150  var name = reader.ReadPascalString(2);
151  var dataLength = (int)reader.ReadUInt32();
152 
153  var dataPaddedLength = Util.RoundUp(dataLength, 2);
154  var endPosition = reader.BaseStream.Position + dataPaddedLength;
155 
156  ImageResource resource = null;
157  var resourceId = (ResourceID)resourceIdInt;
158  switch (resourceId)
159  {
160  case ResourceID.ResolutionInfo:
161  resource = new ResolutionInfo(reader, name);
162  break;
163  case ResourceID.ThumbnailRgb:
164  case ResourceID.ThumbnailBgr:
165  resource = new Thumbnail(reader, resourceId, name, dataLength);
166  break;
167  case ResourceID.AlphaChannelNames:
168  resource = new AlphaChannelNames(reader, name, dataLength);
169  break;
170  case ResourceID.UnicodeAlphaNames:
171  resource = new UnicodeAlphaNames(reader, name, dataLength);
172  break;
173  case ResourceID.VersionInfo:
174  resource = new VersionInfo(reader, name);
175  break;
176  default:
177  resource = new RawImageResource(reader, signature, resourceId, name, dataLength);
178  break;
179  }
180 
181  Util.DebugMessage(reader.BaseStream, "Load, End, ImageResource, {0}",
182  resourceId);
183 
184  // Reposition the reader if we do not consume the full resource block.
185  // This takes care of the even-padding, and also preserves forward-
186  // compatibility in case a resource block is later extended with
187  // additional properties.
188  if (reader.BaseStream.Position < endPosition)
189  reader.BaseStream.Position = endPosition;
190 
191  // However, overruns are definitely an error.
192  if (reader.BaseStream.Position > endPosition)
193  throw new PsdInvalidException("Corruption detected in resource.");
194 
195  return resource;
196  }
197  }

210  public void Set(ImageResource resource)
211  {
212  Predicate<ImageResource> matchId = delegate(ImageResource res)
213  {
214  return res.ID == resource.ID;
215  };
216  var itemIdx = this.FindIndex(matchId);
217  var lastItemIdx = this.FindLastIndex(matchId);
218 
219  if (itemIdx == -1)
220  {
221  Add(resource);
222  }
223  else if (itemIdx != lastItemIdx)
224  {
225  RemoveAll(matchId);
226  Insert(itemIdx, resource);
227  }
228  else
229  {
230  this[itemIdx] = resource;
231  }
232  }
233  }
