325  private static float SlopeAngle(Vector2 start, Vector2 end)
326  {
327  var dir = start - end;
328  dir.Normalize();
329  var dvup = new Vector2(0, 1f);
330  var dvrt = new Vector2(1f, 0);
331 
332  var dr = Vector2.Dot(dir, dvrt);
333  var du = Vector2.Dot(dir, dvup);
334  var cu = Mathf.Acos(du);
335  var sn = dr >= 0 ? 1.0f : -1.0f;
336  var an = cu * Mathf.Rad2Deg * sn;
337 
338  // Adjust angles when direction is parallel to Up Axis.
339  an = (du != 1f) ? an : 0;
340  an = (du != -1f) ? an : -180f;
341  return an;
342  }
