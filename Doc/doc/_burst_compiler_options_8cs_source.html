558  internal string GetOptions(bool isJit, BurstCompileAttribute attr = null, bool isForILPostProcessing = false)
559  {
560  // Add debug to Jit options instead of passing it here
561  // attr.Debug
562 
563  var flagsBuilderOut = new StringBuilder();
564 
565  if (isJit && ((attr?.CompileSynchronously ?? false) || RequiresSynchronousCompilation))
566  {
567  AddOption(flagsBuilderOut, GetOption(OptionJitEnableSynchronousCompilation));
568  }
569 
570  if (isForILPostProcessing)
571  {
572  // IL Post Processing compiles are the only thing set to low priority.
573  AddOption(flagsBuilderOut, GetOption(OptionJitCompilationPriority, CompilationPriority.ILPP));
574  }
575 
576  if (attr != null)
577  {
578  if (attr.FloatMode != FloatMode.Default)
579  {
580  AddOption(flagsBuilderOut, GetOption(OptionFloatMode, attr.FloatMode));
581  }
582 
583  if (attr.FloatPrecision != FloatPrecision.Standard)
584  {
585  AddOption(flagsBuilderOut, GetOption(OptionFloatPrecision, attr.FloatPrecision));
586  }
587 
588  // We disable safety checks for jobs with `[BurstCompile(DisableSafetyChecks = true)]`.
589  if (attr.DisableSafetyChecks)
590  {
591  AddOption(flagsBuilderOut, GetOption(OptionDisableSafetyChecks));
592  }
593 
594  if (attr.Options != null)
595  {
596  foreach (var option in attr.Options)
597  {
598  if (!string.IsNullOrEmpty(option))
599  {
600  AddOption(flagsBuilderOut, option);
601  }
602  }
603  }
604 
605  switch (attr.OptimizeFor)
606  {
607  case OptimizeFor.Default:
608  case OptimizeFor.Balanced:
609  AddOption(flagsBuilderOut, GetOption(OptionOptLevel, 2));
610  break;
611  case OptimizeFor.Performance:
612  AddOption(flagsBuilderOut, GetOption(OptionOptLevel, 3));
613  break;
614  case OptimizeFor.Size:
615  AddOption(flagsBuilderOut, GetOption(OptionOptForSize));
616  AddOption(flagsBuilderOut, GetOption(OptionOptLevel, 3));
617  break;
618  case OptimizeFor.FastCompilation:
619  AddOption(flagsBuilderOut, GetOption(OptionOptLevel, 1));
620  break;
621  }
622  }
623 
624  if (ForceEnableBurstSafetyChecks)
625  {
626  AddOption(flagsBuilderOut, GetOption(OptionGlobalSafetyChecksSetting, GlobalSafetyChecksSettingKind.ForceOn));
627  }
628  else if (EnableBurstSafetyChecks)
629  {
630  AddOption(flagsBuilderOut, GetOption(OptionGlobalSafetyChecksSetting, GlobalSafetyChecksSettingKind.On));
631  }
632  else
633  {
634  AddOption(flagsBuilderOut, GetOption(OptionGlobalSafetyChecksSetting, GlobalSafetyChecksSettingKind.Off));
635  }
636 
637  if (isJit && EnableBurstTimings)
638  {
639  AddOption(flagsBuilderOut, GetOption(OptionJitLogTimings));
640  }
641 
642  if (EnableBurstDebug || (attr?.Debug ?? false))
643  {
644  AddOption(flagsBuilderOut, GetOption(OptionDebugMode));
645  }
646 
647  return flagsBuilderOut.ToString();
648  }
