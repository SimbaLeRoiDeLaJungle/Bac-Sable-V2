356  public override void OnSelectionSceneGUI(GridLayout gridLayout, GameObject brushTarget)
357  {
358  var tilemap = brushTarget.GetComponent<Tilemap>();
359  if (tilemap == null)
360  return;
361 
362  UpdateSelection(tilemap);
363  if (m_SelectionFlagsArray == null || m_SelectionFlagsArray.Length <= 0)
364  return;
365 
366  bool transformFlagsAllEqual = m_SelectionFlagsArray.All(flags => (flags & TileFlags.LockTransform) == (m_SelectionFlagsArray.First() & TileFlags.LockTransform));
367  if (!transformFlagsAllEqual || (m_SelectionFlagsArray[0] & TileFlags.LockTransform) != 0)
368  return;
369 
370  var transformMatrix = m_SelectionMatrices[0];
371  var p = (Vector3)transformMatrix.GetColumn(3);
372  var r = transformMatrix.rotation;
373  var s = transformMatrix.lossyScale;
374  Vector3 selectionPosition = GridSelection.position.position;
375  if (selectionCellCount > 1)
376  {
377  selectionPosition.x = GridSelection.position.center.x;
378  selectionPosition.y = GridSelection.position.center.y;
379  }
380  selectionPosition += tilemap.tileAnchor;
381  var gizmoPosition = tilemap.LocalToWorld(tilemap.CellToLocalInterpolated(selectionPosition + p));
382 
383  EditorGUI.BeginChangeCheck();
384  switch (m_SelectedTransformTool)
385  {
386  case 0:
387  break;
388  case 1:
389  {
390  gizmoPosition = Handles.PositionHandle(gizmoPosition, r);
391  }
392  break;
393  case 2:
394  {
395  r = Handles.RotationHandle(r, gizmoPosition);
396  }
397  break;
398  case 3:
399  {
400  s = Handles.ScaleHandle(s, gizmoPosition, r, HandleUtility.GetHandleSize(gizmoPosition));
401  }
402  break;
403  case 4:
404  {
405  Handles.TransformHandle(ref gizmoPosition, ref r, ref s);
406  }
407  break;
408  default:
409  break;
410  }
411  if (EditorGUI.EndChangeCheck())
412  {
413  RegisterUndo(brushTarget, GridBrushBase.Tool.Select);
414  var offset = tilemap.WorldToLocal(tilemap.LocalToCellInterpolated(gizmoPosition)) - selectionPosition;
415  foreach (var position in GridSelection.position.allPositionsWithin)
416  {
417  if (tilemap.HasTile(position))
418  tilemap.SetTransformMatrix(position, Matrix4x4.TRS(offset, r, s));
419  }
420  InspectorWindow.RefreshInspectors();
421  }

684  private static int GetHash(GridLayout gridLayout, GameObject brushTarget, BoundsInt position, GridBrushBase.Tool tool, GridBrush brush)
685  {
686  int hash = 0;
687  unchecked
688  {
689  hash = hash * 33 + (gridLayout != null ? gridLayout.GetHashCode() : 0);
690  hash = hash * 33 + (brushTarget != null ? brushTarget.GetHashCode() : 0);
691  hash = hash * 33 + position.GetHashCode();
692  hash = hash * 33 + tool.GetHashCode();
693  hash = hash * 33 + (brush != null ? brush.GetHashCode() : 0);
694  }
695 
696  return hash;
697  }
