163  private List<Vector2> Linearize(List<Vector2> src, float pointDistance)
164  {
165  if (src == null) throw new ArgumentNullException("src");
166  if (pointDistance <= kEpsilon) throw new InvalidOperationException("pointDistance " + pointDistance + " is less than epislon " + kEpsilon);
167 
168  var dst = new List<Vector2>();
169 
170  if (src.Count > 0)
171  {
172  var accDistance = 0f;
173  var lastIndex = 0;
174  var lastPoint = src[0];
175 
176  dst.Add(lastPoint);
177 
178  for (var i = 0; i < src.Count; i++)
179  {
180  var nextIndex = NextIndex(i, src.Count);
181  var p0 = src[i];
182  var p1 = src[nextIndex];
183  var edgeDistance = Vector2.Distance(p0, p1);
184 
185  if (accDistance + edgeDistance > pointDistance || nextIndex == 0)
186  {
187  var partialDistance = pointDistance - accDistance;
188  var newPoint = Vector2.Lerp(p0, p1, partialDistance / edgeDistance);
189  var remainingDistance = edgeDistance - partialDistance;
190 
191  //Roll back until we do not intersect any pixel
192  var step = 1f;
193  bool finish = false;
194  while (!finish && IsLineOverImage(newPoint, lastPoint))
195  {
196  partialDistance = Vector2.Distance(p0, newPoint) - step;
197 
198  while (partialDistance < 0f)
199  {
200  if (i > lastIndex + 1)
201  {
202  accDistance -= edgeDistance;
203  --i;
204  p1 = p0;
205  p0 = src[i];
206  edgeDistance = Vector2.Distance(p0, p1);
207  partialDistance += edgeDistance;
208  }
209  else
210  {
211  partialDistance = 0f;
212  finish = true;
213  }
214 
215  remainingDistance = edgeDistance - partialDistance;
216  }
217 
218  newPoint = Vector2.Lerp(p0, p1, partialDistance / edgeDistance);
219  }
220 
221  Debug.Assert(lastIndex <= i, "Generate Outline failed");
222 
223  nextIndex = NextIndex(i, src.Count);
224  if (nextIndex != 0 || !EqualsOrClose(newPoint, p1))
225  {
226  dst.Add(newPoint);
227  lastPoint = newPoint;
228  lastIndex = i;
229  }
230 
231  while (remainingDistance > pointDistance)
232  {
233  remainingDistance -= pointDistance;
234  newPoint = Vector2.Lerp(p0, p1, (edgeDistance - remainingDistance) / edgeDistance);
235  if (!EqualsOrClose(newPoint, lastPoint))
236  {
237  dst.Add(newPoint);
238  lastPoint = newPoint;
239  }
240  }
241  accDistance = remainingDistance;
242  }
243  else
244  {
245  accDistance += edgeDistance;
246  }
247  }
248  }
249  return dst;
250  }

257  private void SmoothPath(Vector2[] path, int iterations, float velocity, float minAngle)
258  {
259  Debug.Assert(iterations > 0f);
260  Debug.Assert(minAngle >= 0f);
261  Debug.Assert(minAngle < 180f);
262 
263  var cosTolerance = Mathf.Cos(minAngle * Mathf.Deg2Rad);
264 
265  for (int iteration = 0; iteration < iterations; ++iteration)
266  for (int i = 0; i < path.Length; ++i)
267  {
268  var prevPoint = path[PreviousIndex(i, path.Length)];
269  var point = path[i];
270  var nextPoint = path[NextIndex(i, path.Length)];
271 
272  var t1 = prevPoint - point;
273  var t2 = nextPoint - point;
274 
275  var dot = Vector2.Dot(t1.normalized, t2.normalized);
276 
277  if (dot > cosTolerance)
278  continue;
279 
280  var w1 = 1f / (point - prevPoint).magnitude;
281  var w2 = 1f / (point - nextPoint).magnitude;
282  var laplacian = (w1 * prevPoint + w2 * nextPoint) / (w1 + w2) - point;
283  point += laplacian * velocity;
284 
285  if (!IsLineOverImage(point, nextPoint) && !IsLineOverImage(point, prevPoint))
286  path[i] = point;
287  }
288  }

329  private IEnumerable<Vector2Int> GetPointsOnLine(int x0, int y0, int x1, int y1)
330  {
331  bool steep = Mathf.Abs(y1 - y0) > Math.Abs(x1 - x0);
332  if (steep)
333  {
334  int t;
335  t = x0; // swap x0 and y0
336  x0 = y0;
337  y0 = t;
338  t = x1; // swap x1 and y1
339  x1 = y1;
340  y1 = t;
341  }
342  if (x0 > x1)
343  {
344  int t;
345  t = x0; // swap x0 and x1
346  x0 = x1;
347  x1 = t;
348  t = y0; // swap y0 and y1
349  y0 = y1;
350  y1 = t;
351  }
352  int dx = x1 - x0;
353  int dy = Mathf.Abs(y1 - y0);
354  int error = dx / 2;
355  int ystep = (y0 < y1) ? 1 : -1;
356  int y = y0;
357  for (int x = x0; x <= x1; x++)
358  {
359  yield return new Vector2Int((steep ? y : x), (steep ? x : y));
360  error = error - dy;
361  if (error < 0)
362  {
363  y += ystep;
364  error += dx;
365  }
366  }
367  yield break;
368  }
