130  void ISpriteEditorDataProvider.SetSpriteRects(SpriteRect[] spriteRects)
131  {
132  if (spriteImportMode != SpriteImportMode.Multiple && spriteImportMode != SpriteImportMode.None && spriteRects.Length == 1)
133  {
134  m_SpriteSingle.CopyFromSpriteRect(spriteRects[0]);
135  }
136  else if (spriteImportMode == SpriteImportMode.Multiple)
137  {
138  Dictionary<GUID, SpriteRect> newSprites = new Dictionary<GUID, SpriteRect>();
139  foreach (var newSprite in spriteRects)
140  {
141  newSprites.Add(newSprite.spriteID, newSprite);
142  }
143  HashSet<long> internalIdUsed = new HashSet<long>();
144  for (int i = m_SpritesMultiple.Count - 1; i >= 0; --i)
145  {
146  var spriteID = m_SpritesMultiple[i].spriteID;
147  if (newSprites.TryGetValue(spriteID, out SpriteRect smd))
148  {
149  m_SpritesMultiple[i].CopyFromSpriteRect(smd);
150  internalIdUsed.Add(m_SpritesMultiple[i].internalID);
151  newSprites.Remove(spriteID);
152  }
153  else
154  {
155  m_SpritesMultiple.RemoveAt(i);
156  }
157  }
158  // Add new ones
159  var nameIdTable = GetSerializedNameFileIdTable(GetSerializedObject());
160  // First pass map by id
161  var values = newSprites.Values.ToArray();
162  foreach (var newSprite in values)
163  {
164  var newSpriteRect = new SpriteDataExt(newSprite);
165  var nameIdPair = nameIdTable.FirstOrDefault(x => x.GetFileGUID() == newSprite.spriteID);
166  if (nameIdPair != null && !internalIdUsed.Contains(nameIdPair.internalID))
167  {
168  newSpriteRect.internalID = nameIdPair.internalID;
169  internalIdUsed.Add(nameIdPair.internalID);
170  m_SpritesMultiple.Add(newSpriteRect);
171  newSprites.Remove(newSprite.spriteID);
172  }
173  }
174  //Second pass map by name
175  foreach (var newSprite in newSprites.Values)
176  {
177  var newSpriteRect = new SpriteDataExt(newSprite);
178  var nameIdPair = nameIdTable.FirstOrDefault(x => x.name == newSprite.name);
179  if (nameIdPair != null && !internalIdUsed.Contains(nameIdPair.internalID))
180  newSpriteRect.internalID = nameIdPair.internalID;
181 
182  internalIdUsed.Add(newSpriteRect.internalID);
183  m_SpritesMultiple.Add(newSpriteRect);
184  }
185  }
186  }

240  void ISpriteEditorDataProvider.Apply()
241  {
242  var so = GetSerializedObject();
243  m_SpriteSingle.Apply(so);
244  var spriteSheetSO = so.FindProperty("m_SpriteSheet.m_Sprites");
245  Dictionary<GUID, SpriteDataExt> newSprites = new Dictionary<GUID, SpriteDataExt>();
246  foreach (var newSprite in m_SpritesMultiple)
247  {
248  newSprites.Add(newSprite.spriteID, newSprite);
249  }
250 
251  if (spriteSheetSO.arraySize > 0)
252  {
253  var validateCurrent = spriteSheetSO.GetArrayElementAtIndex(0);
254  for (int i = 0; i < spriteSheetSO.arraySize; ++i)
255  {
256  var guid = SpriteRect.GetSpriteIDFromSerializedProperty(validateCurrent);
257  // find the GUID in our sprite list and apply to it;
258  if (newSprites.TryGetValue(guid, out SpriteDataExt smd))
259  {
260  smd.Apply(validateCurrent);
261  newSprites.Remove(guid);
262  validateCurrent.Next(false);
263  }
264  else// we can't find it, it is already deleted
265  {
266  spriteSheetSO.DeleteArrayElementAtIndex(i);
267  --i;
268  }
269  }
270  }
271 
272  // Add new ones
273  int newCount = newSprites.Count();
274  if (newCount > 0)
275  {
276  var idx = spriteSheetSO.arraySize;
277  spriteSheetSO.arraySize += newCount;
278  var addCurrent = spriteSheetSO.GetArrayElementAtIndex(idx);
279  foreach (var newSprite in newSprites.Values)
280  {
281  newSprite.Apply(addCurrent);
282  addCurrent.Next(false);
283  }
284  }
