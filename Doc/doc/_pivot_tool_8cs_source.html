156  void TranslatePivotPoint(Vector2 pivot, out SpriteAlignment alignment)
157  {
158  if (new Vector2(k_PivotNormalizedRect.xMin, k_PivotNormalizedRect.yMax) == pivot)
159  alignment = SpriteAlignment.TopLeft;
160  else if(new Vector2(k_PivotNormalizedRect.center.x, k_PivotNormalizedRect.yMax) == pivot)
161  alignment = SpriteAlignment.TopCenter;
162  else if(new Vector2(k_PivotNormalizedRect.xMax, k_PivotNormalizedRect.yMax) == pivot)
163  alignment = SpriteAlignment.TopRight;
164  else if(new Vector2(k_PivotNormalizedRect.xMin, k_PivotNormalizedRect.center.y) == pivot)
165  alignment = SpriteAlignment.LeftCenter;
166  else if(new Vector2(k_PivotNormalizedRect.center.x, k_PivotNormalizedRect.center.y) == pivot)
167  alignment = SpriteAlignment.Center;
168  else if(new Vector2(k_PivotNormalizedRect.xMax, k_PivotNormalizedRect.center.y) == pivot)
169  alignment = SpriteAlignment.RightCenter;
170  else if(new Vector2(k_PivotNormalizedRect.xMin, k_PivotNormalizedRect.yMin) == pivot)
171  alignment = SpriteAlignment.BottomLeft;
172  else if(new Vector2(k_PivotNormalizedRect.center.x, k_PivotNormalizedRect.yMin) == pivot)
173  alignment = SpriteAlignment.BottomCenter;
174  else if(new Vector2(k_PivotNormalizedRect.xMax, k_PivotNormalizedRect.yMin) == pivot)
175  alignment = SpriteAlignment.BottomRight;
176  else
177  alignment = SpriteAlignment.Custom;
178  }

217  Vector2 PivotSlider(Rect sprite, Vector2 pos, GUIStyle pivotDot, GUIStyle pivotDotActive)
218  {
219  int id = GUIUtility.GetControlID(m_SlideHashCode, FocusType.Keyboard);
220 
221  // Convert from normalized space to texture space
222  pos = new Vector2(sprite.xMin + sprite.width * pos.x, sprite.yMin + sprite.height * pos.y);
223 
224  Vector2 screenVal = Handles.matrix.MultiplyPoint(pos);
225 
226  Rect handleScreenPos = new Rect(
227  screenVal.x - pivotDot.fixedWidth * .5f,
228  screenVal.y - pivotDot.fixedHeight * .5f,
229  pivotDotActive.fixedWidth,
230  pivotDotActive.fixedHeight
231  );
232 
233  var evt = Event.current;
234  switch (evt.GetTypeForControl(id))
235  {
236  case EventType.MouseDown:
237  // am I closest to the thingy?
238  if (evt.button == 0 && handleScreenPos.Contains(Event.current.mousePosition) && !evt.alt)
239  {
240  GUIUtility.hotControl = GUIUtility.keyboardControl = id; // Grab mouse focus
241  m_CurrentMousePosition = evt.mousePosition;
242  m_DragStartScreenPosition = evt.mousePosition;
243  Vector2 rectScreenCenter = Handles.matrix.MultiplyPoint(pos);
244  m_DragScreenOffset = m_CurrentMousePosition - rectScreenCenter;
245  evt.Use();
246  EditorGUIUtility.SetWantsMouseJumping(1);
247  }
248  break;
249  case EventType.MouseDrag:
250  if (GUIUtility.hotControl == id)
251  {
252  m_CurrentMousePosition += evt.delta;
253  Vector2 oldPos = pos;
254  Vector3 scrPos = Handles.inverseMatrix.MultiplyPoint(m_CurrentMousePosition - m_DragScreenOffset);
255  pos = new Vector2(scrPos.x, scrPos.y);
256  if (!Mathf.Approximately((oldPos - pos).magnitude, 0f))
257  GUI.changed = true;
258  evt.Use();
259  }
260  break;
261  case EventType.MouseUp:
262  if (GUIUtility.hotControl == id && (evt.button == 0 || evt.button == 2))
263  {
264  GUIUtility.hotControl = 0;
265  evt.Use();
266  EditorGUIUtility.SetWantsMouseJumping(0);
267  }
268  break;
269  case EventType.KeyDown:
270  if (GUIUtility.hotControl == id)
271  {
272  if (evt.keyCode == KeyCode.Escape)
273  {
274  pos = Handles.inverseMatrix.MultiplyPoint(m_DragStartScreenPosition - m_DragScreenOffset);
275  GUIUtility.hotControl = 0;
276  GUI.changed = true;
277  evt.Use();
278  }
279  }
280  break;
281  case EventType.Repaint:
282  EditorGUIUtility.AddCursorRect(handleScreenPos, MouseCursor.Arrow, id);
283 
284  if (GUIUtility.hotControl == id)
285  pivotDotActive.Draw(handleScreenPos, GUIContent.none, id);
286  else
287  pivotDot.Draw(handleScreenPos, GUIContent.none, id);
288 
289  break;
290  }
291 
292  // Convert from texture space back to normalized space
293  pos = new Vector2((pos.x - sprite.xMin) / sprite.width, (pos.y - sprite.yMin) / sprite.height);
294 
295  return pos;
296  }
