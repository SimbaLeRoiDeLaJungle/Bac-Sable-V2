667  for (int i = 0; i < sortedAngleRanges.Count; i++)

678  }

683  angleRangeInfo.start = angleRange.start;
684  angleRangeInfo.end = angleRange.end;
685  angleRangeInfo.order = (uint)angleRange.order;
686  List<int> spriteIndices = new List<int>();
687  foreach (Sprite edgeSprite in angleRange.sprites)
688  {
689  edgeSpriteList.Add(edgeSprite);
690  spriteIndices.Add(edgeSpriteList.Count - 1);
691  }
692  angleRangeInfo.sprites = spriteIndices.ToArray();
693  angleRangeInfoList.Add(angleRangeInfo);
694  }
695  }
696 
697  bool validCornerSpritesFound = false;
698  foreach (CornerSprite cornerSprite in spriteShape.cornerSprites)
699  {
700  if (cornerSprite.sprites[0] != null)
701  {
702  validCornerSpritesFound = true;
703  break;
704  }
705  }
706 
707  if (validCornerSpritesFound)
708  {
709  for (int i = 0; i < spriteShape.cornerSprites.Count; i++)
710  {
711  CornerSprite cornerSprite = spriteShape.cornerSprites[i];
712  cornerSpriteList.Add(cornerSprite.sprites[0]);
713  }
714  }
715 
716  m_EdgeSpriteArray = edgeSpriteList.ToArray();
717  m_CornerSpriteArray = cornerSpriteList.ToArray();
718  m_AngleRangeInfoArray = angleRangeInfoList.ToArray();
719

765  internal int CalculateMaxArrayCount(NativeArray<ShapeControlPoint> shapePoints)
766  {
767  int maxVertexCount = 1024 * 64;
768  bool hasSprites = false;
769  float smallestWidth = 99999.0f;
770 
771  if (null != spriteArray)
772  {
773  foreach (var sprite in m_SpriteArray)
774  {
775  if (sprite != null)
776  {
777  hasSprites = true;
778  float pixelWidth = BezierUtility.GetSpritePixelWidth(sprite);
779  smallestWidth = (smallestWidth > pixelWidth) ? pixelWidth : smallestWidth;
780  }
781  }
782  }
783 
784  // Approximate vertex Array Count. Include Corners and Wide Sprites into account.
785  float smallestSegment = smallestWidth;
786  float shapeLength = BezierUtility.BezierLength(shapePoints, splineDetail, ref smallestSegment) * 4.0f;
787  int adjustShape = shapePoints.Length * 4 * splineDetail;
788  int adjustWidth = hasSprites ? ((int)(shapeLength / smallestSegment) * splineDetail) + adjustShape : 0;
789  adjustShape = optimizeGeometry ? (adjustShape) : (adjustShape * 2);
790  adjustShape = ValidateSpriteShapeTexture() ? adjustShape : 0;
791  maxArrayCount = adjustShape + adjustWidth;
792  maxArrayCount = math.min(maxArrayCount, maxVertexCount);
793  return maxArrayCount;
794  }
