310  private static Type GetType(string TypeName)
311  {
312  var type = Type.GetType(TypeName);
313  if (type != null)
314  return type;
315 
316  if (TypeName.Contains("."))
317  {
318  var assemblyName = TypeName.Substring(0, TypeName.IndexOf('.'));
319  var assembly = Assembly.Load(assemblyName);
320  if (assembly == null)
321  return null;
322  type = assembly.GetType(TypeName);
323  if (type != null)
324  return type;
325  }
326 
327  var currentAssembly = Assembly.GetExecutingAssembly();
328  var referencedAssemblies = currentAssembly.GetReferencedAssemblies();
329  foreach (var assemblyName in referencedAssemblies)
330  {
331  var assembly = Assembly.Load(assemblyName);
332  if (assembly != null)
333  {
334  type = assembly.GetType(TypeName);
335  if (type != null)
336  return type;
337  }
338  }
339  return null;
