305  public static explicit operator double(Int128 val)
306  {
307  const double shift64 = 18446744073709551616.0; //2^64
308  if (val.hi < 0)
309  {
310  if (val.lo == 0)
311  return (double)val.hi * shift64;
312  else
313  return -(double)(~val.lo + ~val.hi * shift64);
314  }
315  else
316  return (double)(val.lo + val.hi * shift64);
317  }
318 
319  //nb: Constructing two new Int128 objects every time we want to multiply longs
320  //is slow. So, although calling the Int128Mul method doesn't look as clean, the
321  //code runs significantly faster than if we'd used the * operator.
322 
323  public static Int128 Int128Mul(Int64 lhs, Int64 rhs)
324  {
325  bool negate = (lhs < 0) != (rhs < 0);
326  if (lhs < 0) lhs = -lhs;
327  if (rhs < 0) rhs = -rhs;
328  UInt64 int1Hi = (UInt64)lhs >> 32;
329  UInt64 int1Lo = (UInt64)lhs & 0xFFFFFFFF;
330  UInt64 int2Hi = (UInt64)rhs >> 32;
331  UInt64 int2Lo = (UInt64)rhs & 0xFFFFFFFF;
332 
333  //nb: see comments in clipper.pas
334  UInt64 a = int1Hi * int2Hi;
335  UInt64 b = int1Lo * int2Lo;
336  UInt64 c = int1Hi * int2Lo + int1Lo * int2Hi;
337 
338  UInt64 lo;
339  Int64 hi;
340  hi = (Int64)(a + (c >> 32));
341 
342  unchecked { lo = (c << 32) + b; }
343  if (lo < b) hi++;
344  Int128 result = new Int128(hi, lo);
345  return negate ? -result : result;
346  }

768  private TEdge ProcessBound(TEdge E, bool LeftBoundIsForward)
769  {
770  TEdge EStart, Result = E;
771  TEdge Horz;
772 
773  if (Result.OutIdx == Skip)
774  {
775  //check if there are edges beyond the skip edge in the bound and if so
776  //create another LocMin and calling ProcessBound once more ...
777  E = Result;
778  if (LeftBoundIsForward)
779  {
780  while (E.Top.Y == E.Next.Bot.Y) E = E.Next;
781  while (E != Result && E.Dx == horizontal) E = E.Prev;
782  }
783  else
784  {
785  while (E.Top.Y == E.Prev.Bot.Y) E = E.Prev;
786  while (E != Result && E.Dx == horizontal) E = E.Next;
787  }
788  if (E == Result)
789  {
790  if (LeftBoundIsForward) Result = E.Next;
791  else Result = E.Prev;
792  }
793  else
794  {
795  //there are more edges in the bound beyond result starting with E
796  if (LeftBoundIsForward)
797  E = Result.Next;
798  else
799  E = Result.Prev;
800  LocalMinima locMin = new LocalMinima();
801  locMin.Next = null;
802  locMin.Y = E.Bot.Y;
803  locMin.LeftBound = null;
804  locMin.RightBound = E;
805  E.WindDelta = 0;
806  Result = ProcessBound(E, LeftBoundIsForward);
807  InsertLocalMinima(locMin);
808  }
809  return Result;
810  }
811 
812  if (E.Dx == horizontal)
813  {
814  //We need to be careful with open paths because this may not be a
815  //true local minima (ie E may be following a skip edge).
816  //Also, consecutive horz. edges may start heading left before going right.
817  if (LeftBoundIsForward) EStart = E.Prev;
818  else EStart = E.Next;
819  if (EStart.Dx == horizontal) //ie an adjoining horizontal skip edge
820  {
821  if (EStart.Bot.X != E.Bot.X && EStart.Top.X != E.Bot.X)
822  ReverseHorizontal(E);
823  }
824  else if (EStart.Bot.X != E.Bot.X)
825  ReverseHorizontal(E);
826  }
827 
828  EStart = E;
829  if (LeftBoundIsForward)
830  {
831  while (Result.Top.Y == Result.Next.Bot.Y && Result.Next.OutIdx != Skip)
832  Result = Result.Next;
833  if (Result.Dx == horizontal && Result.Next.OutIdx != Skip)
834  {
835  //nb: at the top of a bound, horizontals are added to the bound
836  //only when the preceding edge attaches to the horizontal's left vertex
837  //unless a Skip edge is encountered when that becomes the top divide
838  Horz = Result;
839  while (Horz.Prev.Dx == horizontal) Horz = Horz.Prev;
840  if (Horz.Prev.Top.X > Result.Next.Top.X) Result = Horz.Prev;
841  }
842  while (E != Result)
843  {
844  E.NextInLML = E.Next;
845  if (E.Dx == horizontal && E != EStart && E.Bot.X != E.Prev.Top.X)
846  ReverseHorizontal(E);
847  E = E.Next;
848  }
849  if (E.Dx == horizontal && E != EStart && E.Bot.X != E.Prev.Top.X)
850  ReverseHorizontal(E);
851  Result = Result.Next; //move to the edge just beyond current bound
852  }
853  else
854  {
855  while (Result.Top.Y == Result.Prev.Bot.Y && Result.Prev.OutIdx != Skip)
856  Result = Result.Prev;
857  if (Result.Dx == horizontal && Result.Prev.OutIdx != Skip)
858  {
859  Horz = Result;
860  while (Horz.Next.Dx == horizontal) Horz = Horz.Next;
861  if (Horz.Next.Top.X == Result.Prev.Top.X ||
862  Horz.Next.Top.X > Result.Prev.Top.X) Result = Horz.Next;
863  }
864 
865  while (E != Result)
866  {
867  E.NextInLML = E.Prev;
868  if (E.Dx == horizontal && E != EStart && E.Bot.X != E.Next.Top.X)
869  ReverseHorizontal(E);
870  E = E.Prev;
871  }
872  if (E.Dx == horizontal && E != EStart && E.Bot.X != E.Next.Top.X)
873  ReverseHorizontal(E);
874  Result = Result.Prev; //move to the edge just beyond current bound
875  }
876  return Result;
877  }

881  public bool AddPath(Path pg, PolyType polyType, bool Closed)
882  {
883 #if use_lines
884  if (!Closed && polyType == PolyType.ptClip)
885  throw new ClipperException("AddPath: Open paths must be subject.");
886 #else
887  if (!Closed)
888  throw new ClipperException("AddPath: Open paths have been disabled.");
889 #endif
890 
891  int highI = (int)pg.Count - 1;
892  if (Closed) while (highI > 0 && (pg[highI] == pg[0])) --highI;
893  while (highI > 0 && (pg[highI] == pg[highI - 1])) --highI;
894  if ((Closed && highI < 2) || (!Closed && highI < 1)) return false;
895 
896  //create a new edge array ...
897  List<TEdge> edges = new List<TEdge>(highI+1);
898  for (int i = 0; i <= highI; i++) edges.Add(new TEdge());
899 
900  bool IsFlat = true;
901 
902  //1. Basic (first) edge initialization ...
903  edges[1].Curr = pg[1];
904  RangeTest(pg[0], ref m_UseFullRange);
905  RangeTest(pg[highI], ref m_UseFullRange);
906  InitEdge(edges[0], edges[1], edges[highI], pg[0]);
907  InitEdge(edges[highI], edges[0], edges[highI - 1], pg[highI]);
908  for (int i = highI - 1; i >= 1; --i)
909  {
910  RangeTest(pg[i], ref m_UseFullRange);
911  InitEdge(edges[i], edges[i + 1], edges[i - 1], pg[i]);
912  }
913  TEdge eStart = edges[0];
914 
915  //2. Remove duplicate vertices, and (when closed) collinear edges ...
916  TEdge E = eStart, eLoopStop = eStart;
917  for (;;)
918  {
919  //nb: allows matching start and end points when not Closed ...
920  if (E.Curr == E.Next.Curr && (Closed || E.Next != eStart))
921  {
922  if (E == E.Next) break;
923  if (E == eStart) eStart = E.Next;
924  E = RemoveEdge(E);
925  eLoopStop = E;
926  continue;
927  }
928  if (E.Prev == E.Next)
929  break; //only two vertices
930  else if (Closed &&
931  SlopesEqual(E.Prev.Curr, E.Curr, E.Next.Curr, m_UseFullRange) &&
932  (!PreserveCollinear ||
933  !Pt2IsBetweenPt1AndPt3(E.Prev.Curr, E.Curr, E.Next.Curr)))
934  {
935  //Collinear edges are allowed for open paths but in closed paths
936  //the default is to merge adjacent collinear edges into a single edge.
937  //However, if the PreserveCollinear property is enabled, only overlapping
938  //collinear edges (ie spikes) will be removed from closed paths.
939  if (E == eStart) eStart = E.Next;
940  E = RemoveEdge(E);
941  E = E.Prev;
942  eLoopStop = E;
943  continue;
944  }
945  E = E.Next;
946  if ((E == eLoopStop) || (!Closed && E.Next == eStart)) break;
947  }
948 
949  if ((!Closed && (E == E.Next)) || (Closed && (E.Prev == E.Next)))
950  return false;
951 
952  if (!Closed)
953  {
954  m_HasOpenPaths = true;
955  eStart.Prev.OutIdx = Skip;
956  }
957 
958  //3. Do second stage of edge initialization ...
959  E = eStart;
960  do
961  {
962  InitEdge2(E, polyType);
963  E = E.Next;
964  if (IsFlat && E.Curr.Y != eStart.Curr.Y) IsFlat = false;
965  }
966  while (E != eStart);
967 
968  //4. Finally, add edge bounds to LocalMinima list ...
969 
970  //Totally flat paths must be handled differently when adding them
971  //to LocalMinima list to avoid endless loops etc ...
972  if (IsFlat)
973  {
974  if (Closed) return false;
975  E.Prev.OutIdx = Skip;
976  LocalMinima locMin = new LocalMinima();
977  locMin.Next = null;
978  locMin.Y = E.Bot.Y;
979  locMin.LeftBound = null;
980  locMin.RightBound = E;
981  locMin.RightBound.Side = EdgeSide.esRight;
982  locMin.RightBound.WindDelta = 0;
983  for ( ; ; )
984  {
985  if (E.Bot.X != E.Prev.Top.X) ReverseHorizontal(E);
986  if (E.Next.OutIdx == Skip) break;
987  E.NextInLML = E.Next;
988  E = E.Next;
989  }
990  InsertLocalMinima(locMin);
991  m_edges.Add(edges);
992  return true;
993  }
994 
995  m_edges.Add(edges);
996  bool leftBoundIsForward;
997  TEdge EMin = null;
998 
999  //workaround to avoid an endless loop in the while loop below when
1000  //open paths have matching start and end points ...
1001  if (E.Prev.Bot == E.Prev.Top) E = E.Next;
1002 
1003  for (;;)
1004  {
1005  E = FindNextLocMin(E);
1006  if (E == EMin) break;
1007  else if (EMin == null) EMin = E;
1008 
1009  //E and E.Prev now share a local minima (left aligned if horizontal).
1010  //Compare their slopes to find which starts which bound ...
1011  LocalMinima locMin = new LocalMinima();
1012  locMin.Next = null;
1013  locMin.Y = E.Bot.Y;
1014  if (E.Dx < E.Prev.Dx)
1015  {
1016  locMin.LeftBound = E.Prev;
1017  locMin.RightBound = E;
1018  leftBoundIsForward = false; //Q.nextInLML = Q.prev
1019  } else
1020  {
1021  locMin.LeftBound = E;
1022  locMin.RightBound = E.Prev;
1023  leftBoundIsForward = true; //Q.nextInLML = Q.next
1024  }
1025  locMin.LeftBound.Side = EdgeSide.esLeft;
1026  locMin.RightBound.Side = EdgeSide.esRight;
1027 
1028  if (!Closed) locMin.LeftBound.WindDelta = 0;
1029  else if (locMin.LeftBound.Next == locMin.RightBound)
1030  locMin.LeftBound.WindDelta = -1;
1031  else locMin.LeftBound.WindDelta = 1;
1032  locMin.RightBound.WindDelta = -locMin.LeftBound.WindDelta;
1033 
1034  E = ProcessBound(locMin.LeftBound, leftBoundIsForward);
1035  if (E.OutIdx == Skip) E = ProcessBound(E, leftBoundIsForward);
1036 
1037  TEdge E2 = ProcessBound(locMin.RightBound, !leftBoundIsForward);
1038  if (E2.OutIdx == Skip) E2 = ProcessBound(E2, !leftBoundIsForward);
1039 
1040  if (locMin.LeftBound.OutIdx == Skip)
1041  locMin.LeftBound = null;
1042  else if (locMin.RightBound.OutIdx == Skip)
1043  locMin.RightBound = null;
1044  InsertLocalMinima(locMin);
1045  if (!leftBoundIsForward) E = E2;
1046  }
1047  return true;
1048

1876  private void SetWindingCount(TEdge edge)
1877  {
1878  TEdge e = edge.PrevInAEL;
1879  //find the edge of the same polytype that immediately preceeds 'edge' in AEL
1880  while (e != null && ((e.PolyTyp != edge.PolyTyp) || (e.WindDelta == 0))) e = e.PrevInAEL;
1881  if (e == null)
1882  {
1883  PolyFillType pft;
1884  pft = (edge.PolyTyp == PolyType.ptSubject ? m_SubjFillType : m_ClipFillType);
1885  if (edge.WindDelta == 0) edge.WindCnt = (pft == PolyFillType.pftNegative ? -1 : 1);
1886  else edge.WindCnt = edge.WindDelta;
1887  edge.WindCnt2 = 0;
1888  e = m_ActiveEdges; //ie get ready to calc WindCnt2
1889  }
1890  else if (edge.WindDelta == 0 && m_ClipType != ClipType.ctUnion)
1891  {
1892  edge.WindCnt = 1;
1893  edge.WindCnt2 = e.WindCnt2;
1894  e = e.NextInAEL; //ie get ready to calc WindCnt2
1895  }
1896  else if (IsEvenOddFillType(edge))
1897  {
1898  //EvenOdd filling ...
1899  if (edge.WindDelta == 0)
1900  {
1901  //are we inside a subj polygon ...
1902  bool Inside = true;
1903  TEdge e2 = e.PrevInAEL;
1904  while (e2 != null)
1905  {
1906  if (e2.PolyTyp == e.PolyTyp && e2.WindDelta != 0)
1907  Inside = !Inside;
1908  e2 = e2.PrevInAEL;
1909  }
1910  edge.WindCnt = (Inside ? 0 : 1);
1911  }
1912  else
1913  {
1914  edge.WindCnt = edge.WindDelta;
1915  }
1916  edge.WindCnt2 = e.WindCnt2;
1917  e = e.NextInAEL; //ie get ready to calc WindCnt2
1918  }
1919  else
1920  {
1921  //nonZero, Positive or Negative filling ...
1922  if (e.WindCnt * e.WindDelta < 0)
1923  {
1924  //prev edge is 'decreasing' WindCount (WC) toward zero
1925  //so we're outside the previous polygon ...
1926  if (Math.Abs(e.WindCnt) > 1)
1927  {
1928  //outside prev poly but still inside another.
1929  //when reversing direction of prev poly use the same WC
1930  if (e.WindDelta * edge.WindDelta < 0) edge.WindCnt = e.WindCnt;
1931  //otherwise continue to 'decrease' WC ...
1932  else edge.WindCnt = e.WindCnt + edge.WindDelta;
1933  }
1934  else
1935  //now outside all polys of same polytype so set own WC ...
1936  edge.WindCnt = (edge.WindDelta == 0 ? 1 : edge.WindDelta);
1937  }
1938  else
1939  {
1940  //prev edge is 'increasing' WindCount (WC) away from zero
1941  //so we're inside the previous polygon ...
1942  if (edge.WindDelta == 0)
1943  edge.WindCnt = (e.WindCnt < 0 ? e.WindCnt - 1 : e.WindCnt + 1);
1944  //if wind direction is reversing prev then use same WC
1945  else if (e.WindDelta * edge.WindDelta < 0)
1946  edge.WindCnt = e.WindCnt;
1947  //otherwise add to WC ...
1948  else edge.WindCnt = e.WindCnt + edge.WindDelta;
1949  }
1950  edge.WindCnt2 = e.WindCnt2;
1951  e = e.NextInAEL; //ie get ready to calc WindCnt2
1952  }
1953 
1954  //update WindCnt2 ...
1955  if (IsEvenOddAltFillType(edge))
1956  {
1957  //EvenOdd filling ...
1958  while (e != edge)
1959  {
1960  if (e.WindDelta != 0)
1961  edge.WindCnt2 = (edge.WindCnt2 == 0 ? 1 : 0);
1962  e = e.NextInAEL;
1963  }
1964  }
1965  else
1966  {
1967  //nonZero, Positive or Negative filling ...
1968  while (e != edge)
1969  {
1970  edge.WindCnt2 += e.WindDelta;
1971  e = e.NextInAEL;
1972  }
1973  }
1974  }

4162  private static double DistanceFromLineSqrd(IntPoint pt, IntPoint ln1, IntPoint ln2)
4163  {
4164  //The equation of a line in general form (Ax + By + C = 0)
4165  //given 2 points (x¹,y¹) & (x²,y²) is ...
4166  //(y¹ - y²)x + (x² - x¹)y + (y² - y¹)x¹ - (x² - x¹)y¹ = 0
4167  //A = (y¹ - y²); B = (x² - x¹); C = (y² - y¹)x¹ - (x² - x¹)y¹
4168  //perpendicular distance of point (x³,y³) = (Ax³ + By³ + C)/Sqrt(A² + B²)
4169  //see http://en.wikipedia.org/wiki/Perpendicular_distance
4170  double A = ln1.Y - ln2.Y;
4171  double B = ln2.X - ln1.X;
4172  double C = A * ln1.X + B * ln1.Y;
4173  C = A * pt.X + B * pt.Y - C;
4174  return (C * C) / (A * A + B * B);
4175  }
