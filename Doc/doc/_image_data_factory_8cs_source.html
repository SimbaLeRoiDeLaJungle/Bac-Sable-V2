28  public static ImageData Create(Channel channel, byte[] data)
29  {
30  var bitDepth = channel.Layer.PsdFile.BitDepth;
31  ImageData imageData;
32  switch (channel.ImageCompression)
33  {
34  case ImageCompression.Raw:
35  imageData = new RawImage(data, channel.Rect.Size, bitDepth);
36  break;
37 
38  case ImageCompression.Rle:
39  imageData = new RleImage(data, channel.RleRowLengths,
40  channel.Rect.Size, bitDepth);
41  break;
42 
43  case ImageCompression.Zip:
44  // Photoshop treats 32-bit Zip as 32-bit ZipPrediction
45  imageData = (bitDepth == 32)
46  ? CreateZipPredict(data, channel.Rect.Size, bitDepth)
47  : new ZipImage(data, channel.Rect.Size, bitDepth);
48  break;
49 
50  case ImageCompression.ZipPrediction:
51  imageData = CreateZipPredict(data, channel.Rect.Size, bitDepth);
52  break;
53 
54  default:
55  throw new PsdInvalidException("Unknown image compression method.");
56  }
57 
58  // Reverse endianness of multi-byte image data
59  imageData = WrapEndianness(imageData);
60 
61  return imageData;
62  }

64  private static ImageData CreateZipPredict(byte[] data, Size size,
65  int bitDepth)
66  {
67  switch (bitDepth)
68  {
69  case 16:
70  return new ZipPredict16Image(data, size);
71  case 32:
72  return new ZipPredict32Image(data, size);
73  default:
74  throw new PsdInvalidException(
75  "ZIP with prediction is only available for 16 and 32 bit depths.");
76  }
77  }

79  private static ImageData WrapEndianness(ImageData imageData)
80  {
81  // Single-byte image does not require endianness reversal
82  if (imageData.BitDepth <= 8)
83  {
84  return imageData;
85  }
86 
87  // Bytes will be reordered by the compressor, so no wrapper is needed
88  if ((imageData is ZipPredict16Image) || (imageData is ZipPredict32Image))
89  {
90  return imageData;
91  }
92 
93  return new EndianReverser(imageData);
94  }
