512  static bool LineIntersection(float epsilon, float2 p1, float2 p2, float2 p3, float2 p4, ref float2 result)
513  {
514  if (!LineIntersectionTest(epsilon, p1, p2, p3, p4))
515  {
516  return false;
517  }
518 
519  float bx = p2.x - p1.x;
520  float by = p2.y - p1.y;
521  float dx = p4.x - p3.x;
522  float dy = p4.y - p3.y;
523  float bDotDPerp = bx * dy - by * dx;
524  if (math.abs(bDotDPerp) < epsilon)
525  {
526  return false;
527  }
528  float cx = p3.x - p1.x;
529  float cy = p3.y - p1.y;
530  float t = (cx * dy - cy * dx) / bDotDPerp;
531  if ((t >= -epsilon) && (t <= 1.0f + epsilon))

534  result.y = p1.y + t * by;

777  static float2 BezierPoint(float2 st, float2 sp, float2 ep, float2 et, float t)
778  {
779  float2 xt = new float2(t);
780  float2 nt = new float2(1.0f - t);
781  float2 x3 = new float2(3.0f);
782  return (sp * nt * nt * nt) + (st * nt * nt * xt * x3) + (et * nt * xt * xt * x3) + (ep * xt * xt * xt);

785  static float SlopeAngle(float2 dirNormalized)
786  {
787  float2 dvup = new float2(0, 1f);
788  float2 dvrt = new float2(1f, 0);
789 
790  float dr = math.dot(dirNormalized, dvrt);
791  float du = math.dot(dirNormalized, dvup);
792  float cu = math.acos(du);
793  float sn = dr >= 0 ? 1.0f : -1.0f;
794  float an = cu * Mathf.Rad2Deg * sn;
795 
796  // Adjust angles when direction is parallel to Up Axis.
797  an = (du != 1f) ? an : 0;
798  an = (du != -1f) ? an : -180f;
799  return an;
800  }

1104  float2 lt = p1 + pp.tangentLt;

1107 
1108  for (int n = 0; n < splineDetail; ++n)
1109  {

1111  float t = (float) n / fmax;
1112  float2 bp = BezierPoint(rt, p0, p1, lt, t);
1113  xp.position = bp;
1114  spd += math.distance(bp, sp);
1115  m_ContourPoints[ap++] = xp;
1116  sp = bp;
1117  }
1118  sp = p0;
1119 
1120  for (int n = 0; n < splineDetail; ++n)
1121  {
1122  JobContourPoint xp = m_ContourPoints[cap];
1123  cpd += math.distance(xp.position, sp);
1124  xp.ptData.x = smoothInterp ? InterpolateSmooth(cp.cpInfo.x, pp.cpInfo.x, cpd / spd) : InterpolateLinear(cp.cpInfo.x, pp.cpInfo.x, cpd / spd);
1125  m_ContourPoints[cap++] = xp;
1126  sp = xp.position;
1127  }
1128 
1129  }
1130 
1131  // End

1134 
1135  // Create Tessallator if required.
1136  for (int i = 0; i < contourPointCount; ++i)
1137  {

1139  continue;

1151  continue;

1157  vr = new float2(-vr.y, vr.x);
